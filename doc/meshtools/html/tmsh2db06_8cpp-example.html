Example tmsh2db06.cpp (pure triangle)
#include "stdafx.h"
#include "tests_tmsh2db.h"
/*
Input file profile.
This test creates a 2D-layer in the input file profile for several cases of discretization and several depth and growth rate values.
\param[in] filename File name where the initial mesh can be read.
\param[in] depth total depth of the layers.
\param[in] L Number of layers.
\param[in] rho Growth ratio ( (n+1)th layer depth over nth layer depth).
*/
void
TMSH2DB06::run (const char* filename, double d, unsigned L, double rho, const char* mesh_type, bool fill_in)
{
const cm2::element_type FE_type(CM2_FACE_MIX);
cm2::layers_2d::mesher l2dmesher;
cm2::layers_2d::mesher::data_type data;
char str[512];
::strcpy(str, "TMSH2DB06 - LAYER-2D OF THE INPUT PROFILE \"");
::strcat(str, filename);
::strcat(str, "\"");
print_header(str, mesh_type);
// READ IN THE BOUNDARY.
if (read_data(filename, data) < 0)
{
std::cerr << "Error reading \"" << filename << "\".\n";
return;
}
data.depth.resize(data.connectB0.cols(), d);
// LAYER 2D MESH.
l2dmesher.settings.rho = rho;
l2dmesher.settings.nlayers = L;
if (is_T(mesh_type))
l2dmesher.settings.min_Q4_quality = 2.0; // Split all quads.
l2dmesher.run(data);
data.print_info(&display_hdl);
print_footer(data.nods, data.nefs, data.total_time, data.speed, L, d);
// FILL THE INNER DOMAIN WITH TRIANGLES.
if (fill_in)
{
cm2::triamesh_iso::mesher mesher_inside;
fill_inside(mesher_inside, data);
}
// OUTPUT FOR VISUALISATION.
cm2::meshtools::vizir_output("connectB0.mesh", data.pos, data.connectB0, CM2_EDGE2);
cm2::meshtools::vizir_output("connectB1.mesh", data.pos, data.connectB1, CM2_EDGE2);
cm2::meshtools::vizir_output("connectM.mesh", data.pos, data.connectM, FE_type);
}
