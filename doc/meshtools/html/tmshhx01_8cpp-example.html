#include "stdafx.h"
#include "tests_tmshhx.h"
/*
Simple cube.
This is an example of how to use the 3-D unstructured mesher cm2::hexamesh_iso::mesher.
\param[in] L Length of side of the cube.
\param[in] N Number of elements along each side.
*/
void
TMSHHX01::run (double L, unsigned N)
{
cm2::DoubleMat pos;
cm2::UIntMat connectS;
cm2::hexamesh_iso::mesher mesher;
print_header("TMSHHX01 - SIMPLE CUBE");
// Boundary mesh (in matrices pos and connectS).
cm2::meshtools2d::mesh_parallelepiped_Q4(pos, DoubleVec3(0.), DoubleVec3(L), N, N, N, connectS); // Cube centered at (L/2, L/2, L/2).
// Optional visualisation of the boundary mesh.
cm2::meshtools::vizir_output("Q4.mesh", pos, connectS, CM2_FACEQ4);
// The solid mesh.
hexamesh_iso::mesher::data_type data(pos, connectS);
mesher.settings.display_hdl = &display_hdl; // Optional display handler.
mesher.run(data); // Generate the solid mesh.
// Optional visualisation and info output.
data.print_info(&display_hdl);
cm2::meshtools::vizir_output("HX.mesh", data.pos, data.connectM, CM2_SOLID_MIX);
print_footer(data);
}
