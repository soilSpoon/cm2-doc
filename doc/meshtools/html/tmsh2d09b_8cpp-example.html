#include "stdafx.h"
#include "tests_tmsh2d.h"
/*
Plane L-shape domain with circular hole or embedded circular domain.
Same as TMSH2D09 (CM2_FACET3 only) but computes also the boundaries of the Voronoi cells
(approximative-only Voronoi cells).
After the initial meshing, we proceed in several steps to compute the 2-D Voronoi cells:
- Compute the centroid of all boundary edges in matrix `posC` (in same order as `connectE2`). \n
These can be considered as boundary Voronoi points.
- Compute the circumcenters of all triangles in matrix `posV` (in same order as `connectM`). \n
These are the intern Voronoi points.
- Compute the IDs of the boundary edges connected to the boundary nodes (as compressed lists `xadjC` + `adjncyC`). \n
As centroids in `posC`are in same order as boundaries in `connectE2`, these lists are also
valid for the boundary Voronoi points.
- Compute the IDs of the triangles connected to all nodes (as compressed lists `xadjV` + `adjncyV`). \n
As circumcenters in `posV` are in same order as triangles in `connectM`, these lists are also
valid for the Voronoi points around each node.
- Appends the coordinate matrices `posC` and `posV` to `pos` and adjust the indices in the compressed lists.
- Finally, for each node of the mesh we compute:
- The convex hull of the associated Voronoi points (intern + boundary if any).
- Extract the boundary (edges) of the convex hull and append to `connectV`.
\note The initial mesh may not fulfill strictly to the Delaunay criterion (due to boundary constraints,
mesh sizes constraints...) \n
In these cases, the "Voronoi" cells connecting the circumcenters can be non-convex, overlap or
go over the mesh boundaries. \n
The code below produces only _convex_ cells (because we use the convex hull mode of CM2 TriaMesh Iso). \n
But they still can overlap or go over the mesh boundaries.
*/
void
TMSH2D09B::run
(unsigned N1, unsigned N2, bool orient)
{
const element_type FE_type(CM2_FACET3);
const DoubleVec2 P0(-2., -2.);
const DoubleVec2 P1(+4., -2.);
const DoubleVec2 P2(+4., +2.);
const DoubleVec2 P3(+2., +2.);
const DoubleVec2 P4(+2., +8.);
const DoubleVec2 P5(-2., +8.);
const double theta(orient ? 2.*M_PI : -2.*M_PI);
DoubleMat pos;
UIntVec hard_nodes, indices;
UIntMat connectE2, connectM;
cm2::triamesh_iso::mesher mesher;
cm2::triamesh_iso::mesher::data_type data;
print_header("TMSH2D09B - L-SHAPE WITH CIRCULAR HOLE + VORONOI CELLS");
// THE VERTICES.
matvec::push_back(P0, pos); hard_nodes.push_back(0);
matvec::push_back(P1, pos); hard_nodes.push_back(1);
matvec::push_back(P2, pos); hard_nodes.push_back(2);
matvec::push_back(P3, pos); hard_nodes.push_back(3);
matvec::push_back(P4, pos); hard_nodes.push_back(4);
matvec::push_back(P5, pos); hard_nodes.push_back(5);
// THE 1-D MESH.
hard_nodes.push_back(0); // Close the external contour.
meshtools1d::mesh_straight(pos, hard_nodes, N1, /*force_even=>*/ true, indices);
meshtools1d::indices_to_connectE2(indices, connectE2);
indices.clear();
meshtools1d::extrude_rotate(pos, DoubleVec2(1., 0.), DoubleVec2(0.), theta, N2, indices);
indices.back() = indices.front(); // Close the internal circle.
meshtools1d::indices_to_connectE2(indices, connectE2);
// THE 2-D MESH.
{
data.pos = pos;
data.connectB = connectE2;
mesher.run(data);
data.print_info(&display_hdl);
data.extract(pos, connectM);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
// OUTPUT FOR VISUALISATION.
meshtools::vizir_output("T3.mesh", pos, connectM, CM2_FACET3);
// THE VORONOI CELLS.
if (!data.error_raised())
{
const size_t NODS(pos.cols());
DoubleMat posC; // The centroids of edges.
DoubleMat posV; // The Voronoi points (circumcenters).
UIntVec xadjC, adjncyC; // Compressed lists of the connected edges to the boundary nodes.
UIntVec xadjV, adjncyV; // Compressed lists of the connected triangles to all nodes.
UIntMat connectV; // The Voronoi cells (boundaries of all cells).
UIntVec colors; // Cosmetic: a unique color for each Voronoi cell.
// Compute the centroid points of boundary edges.
meshtools::centroids(pos, connectE2, posC, CM2_EDGE2);
// Compute the Voronoi points (circumcenter of the triangles).
meshtools2d::circumcenters(pos, connectM, posV);
// Compute the edges connected to each node (as compressed lists).
meshtools::get_connected_elements(connectE2, xadjC, adjncyC);
// Compute the triangles connected to each node (as compressed lists).
meshtools::get_connected_elements(connectM, xadjV, adjncyV);
// Gather pos, posC and posV (into pos).
meshtools::shift_indices(adjncyC, pos.cols()); // Adjust the indices to new column IDs.
meshtools::shift_indices(adjncyV, pos.cols() + posC.cols()); // Adjust the indices to new column IDs.
pos.push_back(posC);
pos.push_back(posV);
// Compute the convex hull of the Voronoi points around each node.
data.pos = pos;
data.connectB.clear();
mesher.settings.basic_mode = triamesh_iso::mesher::settings_type::CONVEX_HULL_MODE;
mesher.settings.optim_level = 0; // No need to optimize the meshes.
mesher.settings.strict_constraints_flag = false; // Approximate convex hulls are ok here.
for (size_t i = 0; i < NODS; ++i)
{
data.isolated_nodes.copy(adjncyV.sub_vector(xadjV[i], xadjV[i+1] - xadjV[i])); // The IDs of the Voronoi points around node i.
if ((i + 1 < xadjC.size()) && (xadjC[i] < xadjC[i+1])) // Case i is a boundary node.
{
data.isolated_nodes.push_back(unsigned(i)); // Add also node i.
for (size_t ib = xadjC[i]; ib < xadjC[i+1]; ++ib) // Loop over the boundary centroids.
data.isolated_nodes.push_back(adjncyC[ib]);
}
mesher.run(data); // Compute the convex hull.
meshtools::get_mesh_boundaries(data.connectM, data.neighbors,
CM2_FACET3, connectV); // Append boundary of cell.
colors.push_back_n(unsigned(i), connectV.cols() - colors.size()); // Set a color for each Voronoi cell (same as node ID).
}
// OUTPUT FOR VISUALISATION.
meshtools::vizir_output("Vcells.mesh", pos, connectV, CM2_EDGE2, colors);
}
}
