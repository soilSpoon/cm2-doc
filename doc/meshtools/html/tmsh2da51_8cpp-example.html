CM2 TriaMesh Aniso - Simple square with uniform anisotropic metric.
CM2 QuadMesh Aniso - Simple square with uniform anisotropic metric (same result with all_quad_flag = true or false).
#include "stdafx.h"
#include "tests_tmsh2da.h"
/*
This is an example of use of the 2D anisotropic meshers.
Here, the metric is the same on all the hard nodes of the contour,
and by interpolation, the meshers will use a uniform metric all over
the domain.
However, the metric is anisotropic: the target size for the elements is
different along axis X and axis Y.
*/
void
TMSH2DA51::run
(double L1, double L2, unsigned N1, unsigned N2, const char* mesh_type)
{
const element_type FE_type = the_fe_type(mesh_type);
const DoubleVec2 P0(0,0);
const DoubleVec2 P1(L1,0);
const DoubleVec2 P2(L1,L2);
const DoubleVec2 P3(0,L2);
DoubleMat pos;
UIntVec indices;
UIntMat connectE2, connectM;
DoubleMat metrics;
unsigned n;
double D1, D2;
print_header("TMSH2DA51 - RECTANGLE - UNIFORM ANISOTROPIC", mesh_type);
// VERTICES.
matvec::push_back(P0, pos);
matvec::push_back(P1, pos);
matvec::push_back(P2, pos);
matvec::push_back(P3, pos);
// 1D MESHES.
cm2::meshtools1d::mesh_straight(pos, 0, 1, N1, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 1, 2, N2, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 2, 3, N1, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 3, 0, N2, indices);
cm2::meshtools1d::indices_to_connectE2(indices, connectE2);
// METRICS.
D1 = 1. / ((L1 / N1) * (L1 / N1));
D2 = 1. / ((L2 / N2) * (L2 / N2));
metrics.resize(3, pos.cols(), 0.0);
for (size_t i = 0; i < indices.size(); ++i)
{
n = indices[i];
metrics(0, n) = D1; // Mxx
metrics(1, n) = 0.0; // Mxy
metrics(2, n) = D2; // Myy
}
// 2D MESH.
if (::strcmp(mesh_type, "T3") == 0)
{
cm2::triamesh_aniso::mesher the_mesher;
cm2::triamesh_aniso::mesher::data_type data(pos, connectE2);
data.metrics = metrics;
the_mesher.settings.compute_Qh_flag = true;
the_mesher.run(data);
data.print_info(&display_hdl);
data.extract(pos, connectM);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
else
{
cm2::quadmesh_aniso::mesher the_mesher;
cm2::quadmesh_aniso::mesher::data_type data(pos, connectE2);
data.metrics = metrics;
the_mesher.settings.all_quad_flag = (FE_type == CM2_FACEQ4);
the_mesher.settings.compute_Qh_flag = true;
the_mesher.run(data);
data.print_info(&display_hdl);
data.extract(pos, connectM);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
// OUTPUT FOR VISUALISATION.
char filename[64];
::strcpy(filename, mesh_type);
::strcat(filename, ".mesh");
meshtools::vizir_output(filename, pos, connectM, FE_type);
}
