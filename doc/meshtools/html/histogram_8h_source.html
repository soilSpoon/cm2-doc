179template <class Stream>
180void
181print (const histogram& H, Stream& strm)
182{
183 char str[256];
184
185 ::sprintf(str, "Total number of bins : %12d\n", unsigned(H.bins()));
186 strm << str;
187 ::sprintf(str, "Total number of counts : %12d\n", unsigned(H.total_hits()));
188 strm << str;
189 ::sprintf(str, "Number of larger values : %12d\n", unsigned(H.larger()));
190 strm << str;
191 ::sprintf(str, "Number of smaller values : %12d\n", unsigned(H.smaller()));
192 strm << str;
193
194 if (H.total_hits())
195 {
196 ::sprintf(str, "V max : %E\n", H.max_value());
197 strm << str;
198 ::sprintf(str, "V mean : %E\n", H.mean_value());
199 strm << str;
200 ::sprintf(str, "V min : %E\n", H.min_value());
201 strm << str;
202 }
203
204 ::sprintf (str, "\nBin number -- Bin boundaries -- Hits\n\n");
205 strm << str;
206 for (size_t i = H.bins(); i >= 1; --i)
207 {
208 const double lb = H.lower_bin_bound(i-1);
209 const double ub = H.lower_bin_bound(i);
210 const unsigned ht = H.hits(i-1);
211 if ((lb > -DBL_MAX) && (ub < DBL_MAX))
212 ::sprintf(str, "%5d% 16.2f% 16.2f%14d\n", int(i-1), lb, ub, ht);
213 else if (ub < DBL_MAX)
214 ::sprintf(str, "%5d%s% 16.2f%14d\n", int(i-1), " -INF", ub, ht);
215 else if (lb > -DBL_MAX)
216 ::sprintf(str, "%5d% 16.2f%s%14d\n", int(i-1), lb, " +INF", ht);
217 else
218 ::sprintf(str, "%5d%s%s%14d\n", int(i-1), " -INF", " +INF", ht);
219 strm << str;
220 }
221}
