Example of two cylinders intersecting each other repaired by cm2::intersect_t3::mesher.

#include "stdafx.h"
#include "tests_tmsh3di.h"
/*
Intersection between two cylinders with cm2::intersect_t3::mesher.
Intersections are managed between several meshes (up to four in this test case) and between each mesh.
In case of intersection between two faces, the intersecting points are localized on the face of the lowest ID.
So, first faces of the first mesh have the highest priority, then last faces of the first mesh, down to the
last faces of the last mesh.
\param[in] R The cylinders' radius.
\param[in] NR Number of elements along the cylinders' circles.
\param[in] L The cylinders' length.
\param[in] NL Number of elements along the cylinders' height
\param[in] tol Tolerance for the intersections.
Distances below \p tol are considered as null => intersection.
The intersection point is localized on the highest priority face.
*/
void
TMSH3DI01::run
(double R, unsigned NR, double L, unsigned NL, double tol)
{
cm2::DoubleMat pos;
cm2::UIntVec indices;
cm2::UIntMat connectE2, connectS, connectS1, connectS2;
cm2::intersect_t3::mesher intersector;
cm2::triamesh_iso::mesher mesherT3;
unsigned i0, i1;
print_header("TMSH3DI01 - TWO CYLINDERS");
// FIRST CYLINDER.
cm2::meshtools1d::extrude_rotate(pos, DoubleVec3(R, 0., 0.), DoubleVec3(0.), DoubleVec3(0., 0., 2*M_PI), NR, indices);
indices.back() = indices.front();
cm2::meshtools1d::indices_to_connectE2(indices, connectE2);
cm2::triamesh_iso::mesher::data_type dataT3(pos, connectE2);
mesherT3.run(dataT3);
dataT3.extract(pos, connectS1);
cm2::meshtools::copy_mesh(pos, connectS2, connectS1);
cm2::meshtools::rotate(pos, DoubleVec3(0., 0., L/2), DoubleVec3(M_PI, 0., 0.), connectS2);
cm2::matmat::push_back(connectS1, connectS);
cm2::matmat::push_back(connectS2, connectS);
cm2::meshtools1d::flip(connectE2);
cm2::meshtools2d::extrude_translate_T3(pos, connectE2, DoubleVec3(0., 0., L), NL, 0, connectS);
cm2::meshtools::merge(pos, connectS, /*distance=>*/ 1E-6, /*merge_type=>*/ 0); // First cylinder.
i0 = connectS.cols();
// SECOND CYLINDER.
cm2::meshtools::copy_mesh(pos, connectS2, connectS);
cm2::meshtools::translate(pos, DoubleVec3(R, 0., 0.), connectS2);
cm2::meshtools::rotate(pos, DoubleVec3(0., 0., L/2), DoubleVec3(M_PI/2, 0., 0.), connectS2);
cm2::matmat::push_back(connectS2, connectS); // Second cylinder.
i1 = connectS.cols();
// OUTPUT FOR VISUALISATION
cm2::meshtools::vizir_output("M0.mesh", pos, connectS, CM2_FACET3);
// INTERSECTION.
cm2::intersect_t3::mesher::data_type dataI(pos, connectS);
intersector.settings.tolerance = tol;
dataI.xparts.push_back(i0, i1); // Two parts (each one is good).
intersector.run(dataI);
dataI.print_info(&display_hdl);
dataI.extract(pos, connectS);
print_footer(dataI.nods, dataI.nefs, dataI.total_time);
// OUTPUT FOR VISUALISATION.
cm2::meshtools::vizir_output("M1.mesh", pos, connectS, CM2_FACET3, dataI.colors);
// CHECK BOUNDARIES
{
UIntMat V, B;
cm2::meshtools::get_neighbors(connectS, CM2_FACET3, false, V);
cm2::meshtools::get_mesh_boundaries(connectS, V, CM2_FACET3, B);
cm2::meshtools::vizir_output("boundaries.mesh", pos, B, CM2_EDGE2);
}
// REMESH THE INTERSECTION AREA (OPTIONAL).
/*
{
cm2::surfremesh_t3::mesher remesher;
cm2::surfremesh_t3::mesher::data_type dataR(pos, connectS);
remesher.settings.fix_tolerance = -tol; // Absolute tolerance.
remesher.settings.optim_tolerance = -tol; //
remesher.settings.display_hdl = &display_hdl;
for (size_t i = 0; i < dataI.nefs_in_good; ++i) // The non-intersecting faces are the first
dataR.hard_faces.push_back(i); // Do not remesh the non intersecting faces.

dataR.extract(pos, connectS); // The connectivity matrix of the new mesh.
dataR.print_info(&display_hdl);
// OUTPUT OF THE NEW MESH FOR VISUALISATION.
cm2::meshtools::vizir_output("M1_remeshed.mesh", pos, connectS, CM2_FACET3);
}
*/
// THE TET MESH (TO CHECK FOR REMAINING INTERSECTIONS).
{
cm2::tetramesh_iso::mesher tetramesher;
cm2::tetramesh_iso::mesher::data_type dataTH(pos, connectS);
tetramesher.settings.subdomains_forcing = +1;
tetramesher.settings.strict_constraints_flag = true;
tetramesher.settings.refine_flag = false; // No need to refine here.
tetramesher.settings.optim_level = 0; // No need to regularize here.
tetramesher.run(dataTH);
dataTH.print_info(&display_hdl);
}
}
