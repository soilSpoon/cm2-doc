CM2 TetraMesh Iso - Simple sphere with one embedded node at centre with small size (with clipping).
#include "stdafx.h"
#include "tests_tmsh3d.h"
/*
Ball with concentration at centre.
This is an example of how to use the 3-D isotropic unstructured mesher cm2::tetramesh_iso::mesher
with a specific hard node.
\param[in] N Number of elements along 1/8 sectors.
*/
void
TMSH3D18::run (unsigned N)
{
const double R(5.);
cm2::DoubleMat pos(3, 1, 0.); // Point at centre.
cm2::UIntMat connectS;
cm2::tetramesh_iso::mesher mesher;
print_header("TMSH3D18 - FULL SPHERE WITH CONCENTRATION AT CENTRE");
// MESH THE BOUNDARY (T3).
cm2::meshtools2d::mesh_sphere_T3(pos, DoubleVec3(0.), R, N, 2, connectS);
cm2::meshtools::vizir_output("T3.mesh", pos, connectS, CM2_FACET3);
// THE 3-D MESH.
cm2::tetramesh_iso::mesher::data_type data(pos, connectS);
data.isolated_nodes.push_back(0); // Add node #0 (at centre).
data.metrics.push_back(R * (0.025 * (M_PI/4.)) / N); // Specific mesh size only at node #0.
mesher.settings.compute_Qh_flag = true;
mesher.settings.use_default_background_mesh_flag = true;
mesher.run(data);
data.print_info(&display_hdl);
print_footer(data);
// VTK OUTPUT.
cm2::UIntMat connectTH;
cm2::meshtools::clip(data.pos, data.connectM, DoubleVec3(-1E-6), DoubleVec3(0., 0., -1), connectTH); // clip out Z > 0
cm2::meshtools::vizir_output("TH.mesh", data.pos, connectTH, CM2_TETRA4);
}
