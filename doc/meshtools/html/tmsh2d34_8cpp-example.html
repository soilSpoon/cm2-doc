This is an example of how to use the background mesh option with the 2-D meshers cm2::triamesh_iso::mesher and cm2::quadmesh_iso::mesher.

CM2 TriaMesh Iso - Square with metrics defined on a background mesh.
CM2 QuadMesh Iso - Square with metrics defined on a background mesh (all_quad_flag = false).
#include "stdafx.h"
#include "tests_tmsh2d.h"
/*
This example shows how to use the background mesh.
Here, the background mesh is a regular mesh supporting a sinus variation of the metric
in the two directions.
*/
void
TMSH2D34::run
(double L, double h0, double h1, unsigned W, const char* mesh_type)
{
const element_type FE_type(the_fe_type(mesh_type));
const DoubleVec2 P0(-L/2, -L/2);
const DoubleVec2 P1(+L/2, -L/2);
const DoubleVec2 P2(+L/2, +L/2);
const DoubleVec2 P3(-L/2, +L/2);
const unsigned N_BGM = unsigned(0.5*std::max(L/h0, L/h1)); // The discretization for the background mesh.
DoubleMat pos;
DoubleMat::segment_type P;
DoubleVec sizes;
UIntVec indices;
UIntMat connectE2, connectM, BGM;
unsigned n;
double x, h;
cm2::triamesh_iso::mesher triamesher;
cm2::quadmesh_iso::mesher quadmesher;
print_header("TMSH2D34 - SQUARE WITH BACKGROUND MESH", mesh_type);
// THE VERTICES.
matvec::push_back(P0, pos);
matvec::push_back(P1, pos);
matvec::push_back(P2, pos);
matvec::push_back(P3, pos);
// THE 1D MESHES.
cm2::meshtools1d::mesh_straight(pos, 0, 1, h0, h0, true, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 1, 2, h0, h0, true, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 2, 3, h0, h0, true, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 3, 0, h0, h0, true, indices);
cm2::meshtools1d::indices_to_connectE2(indices, connectE2);
// THE BACKGROUND MESH.
indices.clear();
cm2::meshtools1d::mesh_straight(pos, 0, 1, N_BGM, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 1, 2, N_BGM, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 2, 3, N_BGM, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 3, 0, N_BGM, indices);
cm2::meshtools2d::mesh_struct_T3(pos, indices, N_BGM, 2, BGM);
cm2::meshtools::vtk_output("bgm.vtk", pos, BGM, CM2_FACET3);
// THE METRICS ON THE BACKGROUND MESH.
indices.clear();
cm2::meshtools::unique_indices(indices, BGM);
sizes.resize(pos.cols(), 0.0);
for (unsigned i = 0; i < indices.size(); ++i)
{
n = indices[i];
P = pos.seg(n);
x = std::max(::fabs(P[0]), ::fabs(P[1]));
h = ::cos(4.*M_PI*W*x/L) * (h0-h1)/2. + (h0+h1)/2.;
sizes[n] = h;
}
// THE 2D MESH.
if (FE_type == CM2_FACET3)
{
cm2::triamesh_iso::mesher::data_type data(pos, connectE2);
data.background_mesh = BGM;
data.metrics = sizes;
triamesher.settings.compute_Qh_flag = true;
triamesher.run(data);
data.print_info(&display_hdl);
data.extract(pos, connectM);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
else
{
cm2::quadmesh_iso::mesher::data_type data(pos, connectE2);
quadmesher.settings.all_quad_flag = (FE_type == CM2_FACEQ4);
quadmesher.settings.compute_Qh_flag = true;
quadmesher.settings.structured_flag = false;
data.background_mesh = BGM;
data.metrics = sizes;
quadmesher.run(data);
data.print_info(&display_hdl);
data.extract(pos, connectM);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
// OUTPUT FOR VISUALISATION.
char filename[64];
::strcpy(filename, mesh_type);
::strcat(filename, ".mesh");
meshtools::vizir_output(filename, pos, connectM, FE_type);
}
