/*
Simple rectangle.
This is an example of how to use the 2-D isotropic unstructured meshers cm2::triamesh_iso::mesher
and cm2::quadmesh_iso::mesher.
\param[in] N1 Number of elements along side X.
\param[in] N2 Number of elements along side Y.
\param[in] L1 Length of side X of the rectangle.
\param[in] L2 Length of side Y of the rectangle.
\param[in] struct_mode Value to generate unstructured (0) or structured mesh (1, 2).
\param[in] mesh_type Type of mesh (expected "T3", "Q4", or "Q4T3").
*/
void
TMSH2D01::run
(double L1, double L2, unsigned N1, unsigned N2, int struct_mode, const char* mesh_type)
{
const element_type FE_type = the_fe_type(mesh_type);
DoubleMat pos;
UIntVec indices;
UIntMat connectE2, connectM;
print_header("TMSH2D01 - REGULAR RECTANGLE", mesh_type);
// VERTICES.
pos.push_back(DoubleVec2(0.));
pos.push_back(DoubleVec2(L1, 0.));
pos.push_back(DoubleVec2(L1, L2));
pos.push_back(DoubleVec2(0., L2));
// 1D MESHES.
cm2::meshtools1d::mesh_straight(pos, 0, 1, N1, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 1, 2, N2, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 2, 3, N1, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 3, 0, N2, indices);
cm2::meshtools1d::indices_to_connectE2(indices, connectE2);
// 2D MESH.
if (FE_type == CM2_FACET3)
{
cm2::triamesh_iso::mesher mesher;
cm2::triamesh_iso::mesher::data_type data(pos, connectE2);
mesher.settings.structured_pattern = struct_mode;
mesher.settings.compute_Qh_flag = true;
mesher.run(data);
data.print_info(&display_hdl);
data.extract(pos, connectM);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
else
{
cm2::quadmesh_iso::mesher mesher;
cm2::quadmesh_iso::mesher::data_type data(pos, connectE2);
mesher.settings.all_quad_flag = (FE_type == CM2_FACEQ4); // All-quad or quad-dominant mesh.
mesher.settings.structured_flag = (struct_mode > 0);
mesher.settings.compute_Qh_flag = true;
mesher.run(data);
data.print_info(&display_hdl);
data.extract(pos, connectM);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
// OUTPUT FOR VISUALISATION.
char filename[64];
::strcpy(filename, mesh_type);
::strcat(filename, ".mesh");
meshtools::vizir_output(filename, pos, connectM, FE_type);
::strcpy(filename, mesh_type);
::strcat(filename, ".vtk");
meshtools::vtk_output(filename, pos, connectM, FE_type);
}
