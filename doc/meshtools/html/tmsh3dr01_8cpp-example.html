/*
Remeshing of a 3-D surface mesh.
This program takes a surface mesh (read from a STL, Wavefront OBJ or ASCII file)
and remeshes the surface with triangles, quadrangles or a mix of triangles/quadrangles.
\param[in] filename_in File with the initial mesh (STL, Neutral, FEMAP, WaveFront OBJ or standard CM2).
\param[in] min_h Minimum size of along skeleton lines (if possible).
\param[in] max_h Maximum size of along skeleton lines (if possible).
\param[in] target_h Size of the elements inside the patches (if possible).
Usually same value as max_h.
Set to 0 to interpolate sizes from the boundary sizes.
\param[in] optim_tol Tolerance for the nodes mergings and the edge swappings.
\param[in] angle_tol Maximum angle between elements in the same patch.
Angles higher than this threshold are kept in the final mesh (skeleton lines).
\param[in] mesh_type Type of mesh (expected "T3", "Q4", or "Q4T3").
\param[in] simplify_flag Optional. Flag to remove unused nodes from the initial mesh.
*/
void
TMSH3DR01::run
(const char* filename_in, double min_h, double max_h, double target_h,
double optim_tol, double angle_tol, const char* mesh_type,
bool simplify_flag)
{
const element_type FE_type(the_fe_type(mesh_type));
DoubleMat pos0, pos, F0, F;
UIntMat connectM0, connectM;
UIntMat SE0, SE1;
UIntVec colors_in, skedge_colors_in, colors, skedge_colors;
size_t nt(0), nq(0);
char str[512];
::strcpy(str, "TMSH3DR01 - REMESHING OF 3D SURFACE \"");
::strcat(str, filename_in);
::strcat(str, "\" (");
::strcat(str, mesh_type);
::strcat(str, ")");
print_header(str);
// REMESHING OF THE MESH.
if (::strcmp(mesh_type, "T3") == 0) // Remesh with triangles.
{
cm2::surfremesh_t3::mesher remesher;
cm2::surfremesh_t3::mesher::data_type dataR;
// INITIAL MESH.
if (read_data(filename_in, dataR, simplify_flag) < 0)
return; // Error reading file.
// SAVE THE INITIAL MESH.
pos0.copy(dataR.pos);
connectM0.copy(dataR.connectM);
F0.copy(dataR.interpolate_field);
//dataR.hard_nodes.clear();
//dataR.skeleton_edges_in.clear();
//dataR.skeleton_colors_in.clear();
//dataR.colors_in.clear();
//dataR.metrics.clear();
// REMESHING OF THE MESH.
remesher.settings.min_h = min_h;
remesher.settings.max_h = max_h;
remesher.settings.target_h = target_h;
remesher.settings.optim_tolerance = optim_tol;
remesher.settings.patch_angle_tolerance = angle_tol;
remesher.settings.display_hdl = &display_hdl;
remesher.run(dataR);
dataR.extract(pos, connectM); // The connectivity matrix of the new mesh.
SE0 = dataR.skeleton_edges_in;
SE1 = dataR.skeleton_edges;
colors_in = dataR.colors_in;
colors = dataR.colors;
skedge_colors_in = dataR.skeleton_colors_in;
skedge_colors = dataR.skeleton_colors;
F = dataR.interpolate_field;
nt = dataR.nefs_T3;
dataR.print_info(&display_hdl);
print_footer(dataR.nods, dataR.nefs, dataR.total_time, dataR.speed);
//meshtools::simplify(dataR.pos, dataR.connectM);
//dataR.save("remeshed_t3.dat");
}
else // Remesh with quads (and triangles).
{
cm2::surfremesh_q4::mesher remesher;
cm2::surfremesh_q4::mesher::data_type dataR;
// INITIAL MESH.
if (read_data(filename_in, dataR, simplify_flag) < 0)
return; // Error reading file.
// SAVE THE INITIAL MESH.
pos0.copy(dataR.pos);
connectM0.copy(dataR.connectM);
F0.copy(dataR.interpolate_field);
//dataR.hard_nodes.clear();
//dataR.skeleton_edges_in.clear();
//dataR.skeleton_colors_in.clear();
//dataR.colors_in.clear();
//dataR.metrics.clear();
// REMESHING.
remesher.settings.all_quad_flag = (FE_type == CM2_FACEQ4); // All-quad or quad-dominant mesh.
remesher.settings.min_h = min_h;
remesher.settings.max_h = max_h;
remesher.settings.target_h = target_h;
remesher.settings.optim_tolerance = optim_tol;
remesher.settings.patch_angle_tolerance = angle_tol;
remesher.settings.display_hdl = &display_hdl;
remesher.run(dataR);
dataR.extract(pos, connectM); // The connectivity matrix of the new mesh.
SE0 = dataR.skeleton_edges_in;
SE1 = dataR.skeleton_edges;
colors_in = dataR.colors_in;
colors = dataR.colors;
skedge_colors_in = dataR.skeleton_colors_in;
skedge_colors = dataR.skeleton_colors;
F = dataR.interpolate_field;
nq = dataR.nefs_Q4;
nt = dataR.nefs_T3;
dataR.print_info(&display_hdl);
print_footer(dataR.nods, dataR.nefs, dataR.total_time, dataR.speed);
//meshtools::simplify(dataR.pos, dataR.connectM);
//dataR.save("remeshed_q4t3.dat");
}
// OUTPUT OF THE NEW MESH FOR VISUALISATION.
// MEDIT.
cm2::meshtools::vizir_output("SE0.mesh", pos0, SE0, CM2_EDGE2, skedge_colors_in); // The skeleton edges in the initial mesh.
cm2::meshtools::vizir_output("SE1.mesh", pos, SE1, CM2_EDGE2, skedge_colors); // The remeshed skeleton edges.
cm2::meshtools::vizir_output("M0.mesh", pos0, connectM0, CM2_FACE_MIX, colors_in); // The initial mesh.
cm2::meshtools::vizir_output("M1.mesh", pos, connectM, CM2_FACE_MIX, colors); // The final mesh.
// NASTRAN.
{
UIntVec xconnect, refs;
IntVec fe_types;
xconnect.push_back(0u, nq, nq + nt);
fe_types.push_back(int(CM2_FACEQ4), int(CM2_FACET3));
refs.push_back_n(0u, nq + nt);
cm2::meshtools::NASTRAN_output("M1.nas",pos, connectM, xconnect, fe_types, refs);
}
// OBJ.
{
const UIntMat connectQ(connectM.sub_matrix(0, 0, 4, nq));
const UIntMat connectT(connectM.sub_matrix(0, nq, 3, nt));
cm2::meshtools::WavefrontOBJ_output("M1.obj", pos, connectQ, connectT, CM2_FACEQ4, CM2_FACET3);
}
if (!F0.empty())
{
F0.resize(3, F0.cols(), 0.);
F.resize(3, F.cols(), 0.);
cm2::meshtools::vizir_vector("M0.sol", 3, F0); // Works only for 3-D vector fields.
cm2::meshtools::vizir_vector("M1.sol", 3, F); // Works only for 3-D vector fields.
}
// VISUALIZE BOUNDARIES (EDGES WITH != 2 NEIGHBORS).
{
UIntMat V, B;
meshtools::get_neighbors(connectM, FE_type, false, V);
meshtools::get_mesh_boundaries(connectM, V, FE_type, B);
meshtools::vizir_output("BM.mesh", pos, B, CM2_EDGE2);
}
//{
// UIntMat V, B;
// meshtools::get_neighbors(connectM0, FE_type, false, V);
// meshtools::get_mesh_boundaries(connectM0, V, FE_type, B);
// meshtools::vizir_output("BM0.mesh", pos0, B, CM2_EDGE2);
//}
}
