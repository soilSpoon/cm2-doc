99template <class IntersectorE2, class Mesher2D>
100int
101boolean_ops
102 (IntersectorE2& intersector, Mesher2D& mesher2d, typename IntersectorE2::data_type& data, UIntMat& connectMU)
103{
104 const size_t parts(data.xparts.size()); // Nbr of contour parts.
105 const typename IntersectorE2::settings_type SETTINGS_I0(intersector.settings); // Save settings
106 const typename Mesher2D::settings_type SETTINGS_T0(mesher2d.settings); //
107 const cm2::element_type FE2D_TYPE = cm2::element_type(Mesher2D::generated_FE_type);
108 typename Mesher2D::data_type data2D;
109 UIntVec colors_map, color_order, xColor, IDs;
110 UIntMat connect2D;
111 UIntMat connectM0, connectM1, connectMI;
112 timer local_timer;
113 size_t domain_parts(0);
114 unsigned nb0(0), nb1(0), nb(0);
115 int orientation_code(0), ret(0);
116
117 // Prologue.
118 connectMU.clear();
119 if ((FE2D_TYPE != CM2_FACET3) && (FE2D_TYPE != CM2_FACEQ4) && (FE2D_TYPE != CM2_FACE_MIX))
120 return -2; // Error on type of Mesher2D.
121 local_timer.start();
122
123 // Fast return.
124 if (parts == 0)
125 return 0;
126
127 // Repair intersections between parts (or within the part if single).
128 intersector.settings.analysis_code = (parts > 1 ? 0 : 1); // Inter-parts analysis or intra-part analysis.
129 intersector.run(data);
130 intersector.settings = SETTINGS_I0; // Restore settings.
131 if (data.error_raised())
132 return data.error_code;
133
134 if (parts > 0)
135 nb0 = data.xparts[0];
136 if (parts > 1)
137 nb1 = data.xparts[1] - nb0;
138
139 // Apply colors = part IDs.
140 ret = data.set_colors_by_parts(/*reordering_flag=>*/ true);
141 if (ret < 0)
142 return ret; // Memory error?
143
144#ifdef CM2_DBG_FILES
145 meshtools::vizir_output("E2f.mesh", data.pos, data.connectM, CM2_EDGE2, data.colors);
146#endif
147
148 // 2D triangulation for topological analysis.
149 mesher2d.settings.refine_flag = false;
150 mesher2d.settings.optim_level = 0;
151 data2D.pos = data.pos;
152 data2D.connectB = data.connectM;
153 mesher2d.run(data2D);
154 mesher2d.settings = SETTINGS_T0; // Restore settings.
155 data.total_time += data2D.total_time;
156
157#ifdef CM2_DBG_FILES
158 if (!data2D.pathological_boundary_IDs.empty())
159 {
160 UIntMat Cpathos;
161 matmat::push_back(data2D.connectB, data2D.pathological_boundary_IDs, Cpathos);
162 meshtools::vizir_output("E2p.mesh", data2D.pos, Cpathos, CM2_EDGE2);
163 }
164#endif
165
166 if (data2D.error_raised())
167 return data2D.error_code;
168
169 // Re-colorize and reorder subdomains.
170 orientation_code = 2 * data2D.boundary_sgn;
171 ret = data2D.set_colors_by_boundaries(data.connectM, data.colors,
172 /*multicolor_code=>*/ 2,
173 orientation_code,
174 colors_map);
175 if (ret >= 0)
176 data2D.IDs_sorted_by_color(color_order, xColor);
177 if (ret < 0)
178 return ret; // Memory error?
179
180 // Union of contours (without any intern edges).
181 ret = meshtools::get_mesh_boundaries(data2D.connectM, data2D.neighbors, FE2D_TYPE, connectMU);
182 if (ret < 0)
183 return ret; // Memory error?
184 if (data2D.boundary_sgn < 0) // Reorient as initial boundary.
185 meshtools1d::flip(connectMU);
186
187 data.connectM.clear();
188 data.xparts.clear();
189
190 // Number of domain parts (normally two differences + one intersection = 3).
191 if (!xColor.empty())
192 domain_parts = xColor.size() - 1;
193
194 nb = (xColor.empty() ? 0u : xColor.back());
195 if (xColor.size() <= 3) // Case empty parts or empty intersection.
196 xColor.push_back_n(nb, 4 - xColor.size());
197
198 // Part #0 - part #1 (without any intern edges).
199 UIntMat &V(data2D.neighbors); // Recycle matrix data2D.neighbors.
200 connect2D.clear(); V.clear();
201 IDs = color_order.sub_vector(xColor[0], xColor[1] - xColor[0]); // All element IDs with color 0 (domain 0 - domain 1).
202 if (!matmat::push_back(data2D.connectM, IDs, connect2D))
203 return -199; // Memory error.
204 ret = meshtools::get_neighbors(connect2D, FE2D_TYPE,
205 /*accept_multiple_neighbors=>*/ false,
206 V);
207 if (ret >= 0)
208 ret = meshtools::get_mesh_boundaries(connect2D, V, FE2D_TYPE, connectM0);
209 if (ret < 0)
210 return ret; // Memory error?
211 if (data2D.boundary_sgn < 0) // Reorient as initial boundary.
212 meshtools1d::flip(connectM0);
213
214 // Part #1 - part #0 (without any intern edges).
215 connect2D.clear(); V.clear();
216 IDs = color_order.sub_vector(xColor[1], xColor[2] - xColor[1]); // All element IDs with color 1 (domain 1 - domain 0).
217 if (!matmat::push_back(data2D.connectM, IDs, connect2D))
218 return -199; // Memory error.
219 ret = meshtools::get_neighbors(connect2D, FE2D_TYPE,
220 /*accept_multiple_neighbors=>*/ false,
221 V);
222 if (ret >= 0)
223 ret = meshtools::get_mesh_boundaries(connect2D, V, FE2D_TYPE, connectM1);
224 if (ret < 0)
225 return ret; // Memory error?
226 if (data2D.boundary_sgn < 0) // Reorient as initial boundary.
227 meshtools1d::flip(connectM1);
228
229 // Intersection (without any intern edges).
230 connect2D.clear(); V.clear();
231 IDs = color_order.sub_vector(xColor[2], xColor[3] - xColor[2]); // All element IDs with color 2 (intersection).
232 if (!matmat::push_back(data2D.connectM, IDs, connect2D))
233 return -199; // Memory error.
234 ret = meshtools::get_neighbors(connect2D, FE2D_TYPE,
235 /*accept_multiple_neighbors=>*/ false,
236 V);
237 if (ret >= 0)
238 ret = meshtools::get_mesh_boundaries(connect2D, V, FE2D_TYPE, connectMI);
239 if (ret < 0)
240 return ret; // Memory error?
241 if (data2D.boundary_sgn < 0) // Reorient as initial boundary.
242 meshtools1d::flip(connectMI);
243
244 // Fix special cases when intersection is empty.
245 if (connectMI.empty() && (nb0 > 0) && (nb1 > 0))
246 {
247 if (connectM0.cols() + connectM1.cols() == connectMU.cols()) // S0 and S1 disjointed or one fully merged into the other.
248 {
249 if (connectM0.empty() || connectM1.empty()) // One fully merged into the other.
250 connectMI = connectMU;
251 }
252 else if (connectM0.cols() == connectMU.cols() + connectM1.cols()) // S1 fully included in S0.
253 std::swap(connectM1, connectMI);
254 else if (connectM1.cols() == connectMU.cols() + connectM0.cols()) // S0 fully included in S1.
255 std::swap(connectM0, connectMI);
256 }
257
258 data.connectM.push_back(connectM0);
259 data.xparts.push_back(static_cast<unsigned>(data.connectM.cols())); // End of difference domain 0 - domain 1.
260 data.connectM.push_back(connectM1);
261 data.xparts.push_back(static_cast<unsigned>(data.connectM.cols())); // End of difference domain 1 - domain 0.
262 data.connectM.push_back(connectMI);
263 data.xparts.push_back(static_cast<unsigned>(data.connectM.cols())); // End of intersection domain 0, domain 1.
264
265 // Epilogue.
266 data.total_time = local_timer.elapsed_time();
267
268 return int(domain_parts); // Number of parts + intersection.
269}
