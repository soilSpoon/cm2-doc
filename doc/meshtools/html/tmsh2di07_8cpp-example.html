#include "stdafx.h"
#include "tests_tmsh2di.h"
/*
Intersection between two circles.
This is an example of how to use the cm2::intersect_e2::mesher.
\param[in] R1 Radius of the first circle (centered at 0).
\param[in] N1 Number of elements along the first circle.
\param[in] R2 Radius of the second circle (centered at (D2, 0) ).
\param[in] N2 Number of elements along the second circle.
\param[in] D2 Shift of the second circle along X axis.
\param[in] tolerance The intersection/merge tolerance.
*/
void
TMSH2DI07::run
(double R1, unsigned N1, double R2, unsigned N2, double D2, double tolerance)
{
cm2::DoubleMat pos;
cm2::UIntMat connectE20, connectE2;
cm2::UIntVec indices;
print_header("TMSH2DI07 - TWO CIRCLES");
// CONTOUR 1.
cm2::meshtools1d::extrude_rotate(pos, DoubleVec2(R1, 0.), DoubleVec2(0.), 2*M_PI, N1, indices);
// indices.back() = indices.front(); // Close the contour.
cm2::meshtools1d::indices_to_connectE2(indices, connectE2);
// CONTOUR 2.
indices.clear();
cm2::meshtools1d::extrude_rotate(pos, DoubleVec2(R2 + D2, 0.), DoubleVec2(D2, 0.), 2*M_PI, N1, indices);
// indices.back() = indices.front(); // Close the contour.
cm2::meshtools1d::indices_to_connectE2(indices, connectE2);
connectE20.copy(connectE2); // Save initial mesh.
// 2D INTERSECTION.
cm2::intersect_e2::mesher intersector;
cm2::intersect_e2::mesher::data_type data(pos, connectE2);
intersector.settings.tolerance = tolerance;
intersector.run(data);
data.extract(pos, connectE2);
data.print_info(&display_hdl);
print_footer(data.nods, data.nefs, data.total_time);
// OUTPUT FOR VISUALISATION.
cm2::meshtools::vizir_output("initial_contour.mesh", pos, connectE20, CM2_EDGE2);
cm2::meshtools::vizir_output("repaired_contour.mesh", pos, connectE2, CM2_EDGE2);
// REMESH LINES.
{
const double target_h(2.*M_PI*std::min(R1, R2) / 32); // A new mesh size (uniform).
cm2::meshtools1d::remesh_lines(pos, connectE2, target_h);
cm2::meshtools::vizir_output("final_contour.mesh", pos, connectE2, CM2_EDGE2);
}
// 2D MESH.
pos.resize(2, pos.cols());
cm2::triamesh_iso::mesher mesherT3;
cm2::triamesh_iso::mesher::data_type dataT3(pos, connectE2);
mesherT3.settings.subdomains_forcing = +1;
mesherT3.run(dataT3);
dataT3.print_info(&display_hdl);
// OUTPUT FOR VISUALISATION.
cm2::meshtools::vizir_output("T3.mesh", dataT3.pos, dataT3.connectM, CM2_FACET3);
}
