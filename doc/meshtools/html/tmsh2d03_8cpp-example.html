This is an example of how to prescribe a point with the 2-D meshers cm2::triamesh_iso::mesher and cm2::quadmesh_iso::mesher.
The field data.metrics is also used to show how to get a finer mesh near this point.

CM2 TriaMesh Iso - Simple square with input 40 boundary edges and one embedded node at centre with small size.
CM2 QuadMesh Iso - Simple square with input 40 boundary edges and one embedded node at centre with small size (all_quad_flag = false).
CM2 QuadMesh Iso - Simple square with input 40 boundary edges and one embedded node at centre with small size (all_quad_flag = true).
#include "stdafx.h"
#include "tests_tmsh2d.h"
/*
This example shows how to get a finer mesh near a specific node with the 2D meshers
by the use of the data_type::isolated_nodes and the data_type::metrics fields.
In addition, the effect of the flag parameters_type::use_default_background_mesh_flag
can be seen.
*/
void
TMSH2D03::run
(double L, unsigned N, bool use_background_mesh, const char* mesh_type)
{
const cm2::element_type FE_type = the_fe_type(mesh_type);
cm2::DoubleMat pos;
cm2::UIntVec indices;
cm2::DoubleVec sizes;
cm2::UIntMat connectE2, connectM;
cm2::triamesh_iso::mesher triamesher;
cm2::quadmesh_iso::mesher quadmesher;
print_header("TMSH2D03 - SQUARE WITH CONCENTRATION AT CENTRE", mesh_type);
// THE VERTICES.
pos.push_back(DoubleVec2(0.));
pos.push_back(DoubleVec2(L, 0.));
pos.push_back(DoubleVec2(L, L));
pos.push_back(DoubleVec2(0., L));
pos.push_back(DoubleVec2(L/2, L/2));
// THE 1D MESH.
cm2::meshtools1d::mesh_straight(pos, 0, 1, N, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 1, 2, N, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 2, 3, N, indices); indices.pop_back();
cm2::meshtools1d::mesh_straight(pos, 3, 0, N, indices);
cm2::meshtools1d::indices_to_connectE2(indices, connectE2);
// METRICS.
sizes.resize(5, 0.);
sizes[4] = 0.1 * L/N; // Mesh size for the node at center (node #4)
// Default value for all other nodes.
// THE 2D MESH.
if (FE_type == CM2_FACET3)
{
cm2::triamesh_iso::mesher::data_type data(pos, connectE2);
data.isolated_nodes.push_back(4); // The node at center.
data.metrics = sizes;
triamesher.settings.use_default_background_mesh_flag = use_background_mesh;
triamesher.settings.compute_Qh_flag = true;
triamesher.run(data);
data.print_info(&display_hdl);
data.extract(pos, connectM);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
else
{
cm2::quadmesh_iso::mesher::data_type data(pos, connectE2);
data.isolated_nodes.push_back(4); // The node at center.
data.metrics = sizes;
quadmesher.settings.all_quad_flag = (FE_type == CM2_FACEQ4);
quadmesher.settings.use_default_background_mesh_flag = use_background_mesh;
quadmesher.settings.compute_Qh_flag = true;
quadmesher.run(data);
data.print_info(&display_hdl);
data.extract(pos, connectM);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
// OUTPUT FOR VISUALISATION.
char filename[64];
::strcpy(filename, mesh_type);
::strcat(filename, ".mesh");
cm2::meshtools::vizir_output(filename, pos, connectM, FE_type);
}
