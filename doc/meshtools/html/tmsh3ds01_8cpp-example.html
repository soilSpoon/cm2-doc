#include "stdafx.h"
#include "tests_tmsh3ds.h"
/*
Meshing of a CAD model (STEP, IGES, BRep) with cm2::surfmesh_t3::mesher or cm2::surfmesh_q4::mesher.
This program meshes a CAD model with triangles, quadrangles or a mix of triangles/quadrangles.
\param[in] CAD_filename The CAD file (STEP, IGES, BRep).
\param[in] target_h The mesh size.
\param[in] mesh_type Type of mesh (expected "T3", "Q4", "Q4T3", "T6", "Q8", "Q8T6", "Q9", "Q9T6").
\param[in] max_chordal_error Maximum chordal error allowed. <br>
The \p target_h size will be reduced locally to limit the chordal error below this threshold.
\param[in] chordal_control_type The chordal control type.
# 0: No curvature is computed and the chordal error control is disabled.
# 1: Approximate curvature (least-square method using the local bases). <br>
Isotropic chordal control: element sizes are limited by the same (isotropic) chordal value.
# 2: Approximate curvature (least-square method using the local bases). <br>
Anisotropic chordal control: element sizes are limited by two different (anisotropic) chordal values
along the two principal curvature directions.
# 3: Exact curvature (provided by Surface::get_curvatures). <br>
Isotropic chordal control: element sizes are limited by the same (isotropic) chordal value.
# 4: Exact curvature (provided by Surface::get_curvatures). <br>
Anisotropic chordal control: element sizes are limited by two different (anisotropic) chordal values
along the two principal curvature directions.
\param[in] min_h Security minimum mesh size to avoid too fine elements.
\param[in] fix_tol Fix distance tolerance. Vertices within this distance and curves within this distance are merged.
\param[in] max_gradation The max mesh gradation. The smallest, the smoothest size gradations.
\param[in] optim_level The optimization level (from 0 to 10).
*/
void
TMSH3DS01::run (const char* CAD_filename, double target_h, const char* mesh_type,
double max_chordal_error, unsigned chordal_control_type,
double min_h, double fix_tol, double max_gradation, unsigned optim_level)
{
const element_type FE_type(the_fe_type(mesh_type));
const element_type E_FE_type(the_boundary_fe_type(mesh_type));
DoubleMat pos;
UIntMat connectE, connectM, connectMlin;
UIntVec colorsE, colorsM, colorsMlin;
size_t nt(0), nq(0);
char str[512];
::strcpy(str, "TMSH3DS01 - MESHING OF CAD MODEL \"");
::strcat(str, CAD_filename);
::strcat(str, "\" (");
::strcat(str, mesh_type);
::strcat(str, ")");
print_header(str);
// MESHING WITH TRIANGLES.
if (is_T(mesh_type))
{
cm2::surfmesh_t3::mesher the_mesher;
cm2::surfmesh_t3::mesher::data_type data;
the_mesher.settings.target_h = target_h;
the_mesher.settings.min_h = min_h;
the_mesher.settings.fix_tolerance = fix_tol;
the_mesher.settings.max_chordal_error = max_chordal_error;
the_mesher.settings.max_gradation = max_gradation;
the_mesher.settings.chordal_control_type = chordal_control_type;
the_mesher.settings.optim_level = optim_level;
the_mesher.settings.high_order_type = (is_quadratic(mesh_type) ? 1 : 0);
the_mesher.settings.display_hdl = &display_hdl;
//the_mesher.settings.max_gradation = 0.2;
//data.surface_H.resize(40, 0.);
//data.surface_H[2] = data.surface_H[5] = 1.;
//data.curve_H.resize(109, 0.);
//data.curve_H[38] = 0.5;
//data.vertex_H.resize(73, 0.);
//data.vertex_H[66] = 0.25;
the_mesher.run(CAD_filename, data);
data.extract(pos, connectM); // The connectivity matrix of the mesh.
connectE = data.connectE;
colorsM = data.colors;
colorsE = data.colorsE;
nt = data.nefs_T;
data.print_info(&display_hdl);
print_footer(data.nods, data.nefs, data.mesh_time, data.speed);
if (!data.failed.empty())
{
std::cout << "Surface(s) that could not be meshed: \n";
vecio::write(std::cout, data.failed);
}
}
// MESHING WITH QUADS (PURE OR MIXED).
else
{
cm2::surfmesh_q4::mesher the_mesher;
cm2::surfmesh_q4::mesher::data_type data;
the_mesher.settings.target_h = target_h;
the_mesher.settings.min_h = min_h;
the_mesher.settings.fix_tolerance = fix_tol;
the_mesher.settings.max_chordal_error = max_chordal_error;
the_mesher.settings.max_gradation = max_gradation;
the_mesher.settings.chordal_control_type = chordal_control_type;
the_mesher.settings.all_quad_flag = is_Q(mesh_type); // All-quad or quad-dominant mesh.
the_mesher.settings.optim_level = optim_level;
if ((::strcmp(mesh_type, "Q8") == 0) || (::strcmp(mesh_type, "Q8T6") == 0))
the_mesher.settings.high_order_type = 1; // Quadratic elements (serendipity).
if ((::strcmp(mesh_type, "Q9") == 0) || (::strcmp(mesh_type, "Q9T6") == 0))
the_mesher.settings.high_order_type = 2; // Quadratic elements (full).
the_mesher.settings.display_hdl = &display_hdl;
//the_mesher.settings.max_gradation = 0.2;
//data.surface_H.resize(40, 0.);
//data.surface_H[2] = data.surface_H[5] = 1.;
//data.curve_H.resize(109, 0.);
//data.curve_H[38] = 0.5;
//data.vertex_H.resize(73, 0.);
//data.vertex_H[66] = 0.25;
the_mesher.run(CAD_filename, data);
data.extract(pos, connectM); // The connectivity matrix of the mesh.
connectE = data.connectE;
colorsM = data.colors;
colorsE = data.colorsE;
nq = data.nefs_Q;
nt = data.nefs_T;
data.print_info(&display_hdl);
print_footer(data.nods, data.nefs, data.mesh_time, data.speed);
if (!data.failed.empty())
{
std::cout << "Surface(s) that could not be meshed: \n";
vecio::write(std::cout, data.failed);
}
}
// VISUALISATION OUTPUTS.
char out_filename0[512], out_filename[512];
if (FE_type == cm2::CM2_FACET3)
::strcpy(out_filename0, "t3");
else if (FE_type == cm2::CM2_FACEQ4)
::strcpy(out_filename0, "q4");
else if (FE_type == cm2::CM2_FACE_MIX)
::strcpy(out_filename0, "q4t3");
else if (FE_type == cm2::CM2_FACET6)
::strcpy(out_filename0, "t6");
else if (FE_type == cm2::CM2_FACEQ8)
::strcpy(out_filename0, "q8");
else if (FE_type == cm2::CM2_FACEQ9)
::strcpy(out_filename0, "q9");
else if (::strcmp(mesh_type, "Q8T6") == 0)
::strcpy(out_filename0, "q8t6");
else if (::strcmp(mesh_type, "Q9T6") == 0)
::strcpy(out_filename0, "q9t6");
else
strcpy(out_filename0, ""); // Error.
if ((FE_type == cm2::CM2_FACEQ8) || (::strcmp(mesh_type, "Q8T6") == 0)) // MEDIT/VIZIR doesn't support Q8 elements.
meshtools2d::convert_into_linear(connectM);
// MEDIT.
::strcpy(out_filename, out_filename0);
::strcat(out_filename, ".mesh");
if (is_quadratic(mesh_type))
cm2::meshtools::vizir_output(out_filename, pos, connectM, CM2_FACE_QUAD_MIX, colorsM);
else
cm2::meshtools::vizir_output(out_filename, pos, connectM, CM2_FACE_MIX, colorsM);
cm2::meshtools::vizir_output("curves.mesh", pos, connectE, E_FE_type, colorsE); // The edges mesh.
// NASTRAN.
{
UIntVec xconnect, refs;
IntVec fe_types;
::strcpy(out_filename, out_filename0);
::strcat(out_filename, ".nas");
xconnect.push_back(0, nq, nq + nt);
if (is_quadratic(mesh_type))
fe_types.push_back(int(CM2_FACEQ9), int(CM2_FACET6));
else
fe_types.push_back(int(CM2_FACEQ4), int(CM2_FACET3));
refs.push_back_n(0u, nq + nt);
cm2::meshtools::NASTRAN_output(out_filename, pos, connectM, xconnect, fe_types, refs);
}
// OBJ (as linear elements)
{
UIntMat connectQ, connectT;
meshtools2d::convert_into_linear(connectM);
connectQ = connectM.sub_matrix(0, 0, 4, nq);
connectT = connectM.sub_matrix(0, nq, 3, nt);
::strcpy(out_filename, out_filename0);
::strcat(out_filename, ".obj");
cm2::meshtools::WavefrontOBJ_output(out_filename, pos, connectQ, connectT, CM2_FACEQ4, CM2_FACET3);
}
// VISUALIZE BOUNDARIES (EDGES WITH != 2 NEIGHBORS).
{
UIntMat V, B;
meshtools::get_neighbors(connectM, CM2_FACE_MIX, false, V);
meshtools::get_mesh_boundaries(connectM, V, CM2_FACE_MIX, B);
meshtools::vizir_output("BM.mesh", pos, B, CM2_EDGE2);
}
}
