/*
Simple cube.
This is an example of how to use the 3-D isotropic unstructured mesher cm2::tetramesh_iso::mesher.
The boundary triangle mesh is UJ-structured.
\param[in] L Length of side of the cube.
\param[in] N Number of elements along each side.
\param[in] C Centre of the cube (same value in X, Y and Z). Default 0.
*/
void
TMSH3D01::run (double L, unsigned N, double C)
{
cm2::DoubleMat pos;
cm2::UIntMat connectS;
cm2::tetramesh_iso::mesher mesher;
print_header("TMSH3D01 - CUBE - UJ-STRUCTURED BOUNDARY MESH");
// Boundary mesh (in matrices pos and connectS).
cm2::meshtools2d::mesh_parallelepiped_T3(pos, DoubleVec3(C-0.5*L), DoubleVec3(C+0.5*L), // Cube centered at (C, C, C).
N, N, N, /*pattern=>*/ 2, connectS);
// Optional visualisation of the surface mesh.
cm2::meshtools::vizir_output("T3.mesh", pos, connectS, CM2_FACET3);
// The solid mesh.
cm2::tetramesh_iso::mesher::data_type data(pos, connectS);
mesher.settings.compute_Qh_flag = true; // Optional flag to compute length quality of edges.
mesher.settings.display_hdl = &display_hdl; // Optional display handler.
mesher.run(data); // Generate the solid mesh.
// Optional visualisation and info output.
data.print_info(&display_hdl);
cm2::meshtools::vizir_output("TH.mesh", data.pos, data.connectM, CM2_TETRA4);
print_footer(data);
}
