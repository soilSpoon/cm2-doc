/*
Torus.
This is an example of how to use the 3-D isotropic unstructured mesher cm2::tetramesh_iso::mesher.
The boundary triangle mesh is UJ-structured.
\param[in] RA First radius (mid-line of the torus).
\param[in] RB Second radius (perpendicular to the mid-line of the torus).
\param[in] NA Number of elements along the mid-line.
\param[in] NB Number of elements perpendicular to the mid-line.
*/
void
TMSH3D05::run (double RA, double RB, unsigned NA, unsigned NB)
{
DoubleMat pos;
UIntVec indices;
UIntMat connectE, connectS, connectTH4;
tetramesh_iso::mesher mesher;
print_header("TMSH3D05 - TORUS - UJ-STRUCTURED BOUNDARY MESH");
// THE BOUNDARY TRIANGLE MESH (UJ PATTERN)
meshtools1d::extrude_rotate(pos, DoubleVec3(RA+RB, 0., 0.), DoubleVec3(RA, 0., 0.), DoubleVec3(0,2*M_PI,0), NB, indices);
meshtools1d::indices_to_connectE2(indices, connectE);
meshtools2d::extrude_rotate_T3(pos, connectE, DoubleVec3(0.), DoubleVec3(0., 0., 2*M_PI), NA, 2, connectS);
meshtools::merge(pos, connectS, /*distance=>*/ 1E-6, /*merge_type=>*/ 0);
// THE 3-D MESH.
tetramesh_iso::mesher::data_type dataTH(pos, connectS);
mesher.run(dataTH);
dataTH.print_info(&display_hdl);
// OUTPUT.
meshtools::vizir_output("T3.mesh", pos, connectS, CM2_FACET3);
meshtools::vizir_output("TH.mesh", dataTH.pos, dataTH.connectM, CM2_TETRA4);
print_footer(dataTH);
}
