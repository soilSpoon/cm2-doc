#include "stdafx.h"
#include "tests_tmsh2di.h"
/*
Intersection between contours followed by a line remeshing.
This is an example of how to use the cm2::intersect_e2::mesher and cm2::meshtools1d::remesh_lines.
\param[in] filename File with a edge mesh (several lines, contours, 2D or 3D).
\param[in] mesh_type Type of mesh (expected "T3", "Q4", or "Q4T3").
\param[in] tol Tolerance for the intersections.
Distances below \p tol are considered as null => intersection.
The intersection point is localized on the highest priority face.
*/
void
TMSH2DI50::run
(const char* filename, const char* mesh_type, double tol)
{
const cm2::element_type FE_type(the_fe_type(mesh_type));
cm2::intersect_e2::mesher intersector;
cm2::intersect_e2::mesher::data_type dataI;
cm2::DoubleMat pos;
cm2::UIntMat connectB;
cm2::UIntVec hard_nodes;
char str[512];
::strcpy(str, "TMSH2DI50 - \"");
::strcat(str, filename);
::strcat(str, "\"");
print_header(str, mesh_type);
// READ THE INPUT DATA.
if (read_data(filename, dataI) < 0)
return;
cm2::meshtools::vizir_output("initial_contour.mesh", dataI.pos, dataI.connectM, CM2_EDGE2); // Some outputs.
// FIX INTERSECTIONS.
hard_nodes.copy(dataI.isolated_nodes);
intersector.settings.tolerance = tol;
intersector.run(dataI);
dataI.print_info(&display_hdl); // Some outputs.
dataI.extract(pos, connectB);
print_footer(dataI.nods, dataI.nefs, dataI.total_time);
// Isolated nodes have lowest priority for intersect_e2 (replaced with new nodes).
// If we want to keep these isolated nodes, we need to replace them back.
if (!hard_nodes.empty())
{
cm2::meshtools::node_detector detector(pos, connectB);
cm2::UIntVec nodes_map(pos.cols());
for (size_t i = 0; i < pos.cols(); ++i)
nodes_map[i] = static_cast<unsigned>(i); // Iota.
for (size_t i = 0; i < hard_nodes.size(); ++i)
{
const unsigned Ni = hard_nodes[i];
const unsigned Nj = detector.find_closest(Ni); // The node which replaces Ni in connectB.
if (Nj != CM2_NONE)
nodes_map[Nj] = Ni;
}
cm2::meshtools::change_indices(connectB, nodes_map);
}
cm2::meshtools::vizir_output("repaired_contour.mesh", pos, connectB, CM2_EDGE2);
// REMESH LINES.
double target_h(0.);
if (!dataI.error_raised())
{
cm2::UIntVec colors;
cm2::meshtools1d::length(pos, connectB, target_h);
target_h /= connectB.cols(); // Arbitrary new mesh size (uniform).
// target_h *= 4;
cm2::meshtools1d::remesh_lines(pos, connectB, target_h, cm2::DoubleVec(), colors, hard_nodes, cm2::UIntVec(),
/*force_even_flag=>*/ (FE_type == CM2_FACEQ4));
cm2::meshtools::vizir_output("final_contour.mesh", pos, connectB, CM2_EDGE2, colors);
}
// target_h *= 10.;
// THE 2D MESH
if (!dataI.error_raised())
{
if (FE_type == CM2_FACET3)
{
cm2::triamesh_iso::mesher the_mesher;
cm2::triamesh_iso::mesher::data_type dataT3(pos, connectB);
dataT3.isolated_nodes = hard_nodes;
//the_mesher.settings.target_metric = target_h;
//the_mesher.settings.max_gradation = 0.25;
//the_mesher.settings.refine_flag = false;
//the_mesher.settings.optim_level = 0;
the_mesher.settings.display_hdl = &display_hdl;
the_mesher.run(dataT3);
dataT3.print_info(&display_hdl);
cm2::meshtools::vizir_output("T3.mesh", dataT3.pos, dataT3.connectM, FE_type); // Visualization.
}
else
{
cm2::quadmesh_iso::mesher the_mesher;
cm2::quadmesh_iso::mesher::data_type dataQ4(pos, connectB);
dataQ4.isolated_nodes = hard_nodes;
// the_mesher.settings.target_metric = target_h;
// the_mesher.settings.max_gradation = 0.25;
the_mesher.settings.all_quad_flag = (FE_type == CM2_FACEQ4);
the_mesher.settings.display_hdl = &display_hdl;
the_mesher.run(dataQ4);
dataQ4.print_info(&display_hdl);
if (FE_type == CM2_FACEQ4) // Visualization.
meshtools::vizir_output("Q4.mesh", dataQ4.pos, dataQ4.connectM, FE_type);
else
meshtools::vizir_output("Q4T3.mesh", dataQ4.pos, dataQ4.connectM, FE_type);
}
}
}
