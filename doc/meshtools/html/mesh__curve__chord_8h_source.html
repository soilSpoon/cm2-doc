73template <class ForwardIter>
74bool
75__is_sorted (ForwardIter first, ForwardIter last)
76{
77 if (first == last)
78 return true;
79 ForwardIter next(first);
80 for (++next; next != last; first = next, ++next)
81 {
82 if (*next < *first)
83 return false;
84 }
85 return true;

233template <class Curve>
234int
235mesh_curve_param
236 (const Curve& C,
237 DoubleMat& pos, const UIntVec& nodes, DoubleVec& pos1d, DoubleVec& H, UIntMat& connectE,
238 double target_h, bool force_even, unsigned min_n, unsigned max_n,
239 double max_chordal_error, double min_h, unsigned chordal_control_type, double max_gradation,
240 unsigned high_order_type = 0, unsigned high_order_mode = 2,
241 bool new_bgm_flag = true, unsigned max_bgm_remeshings = 4, bool update_metrics_flag = true)
242{
243 const size_t NP(pos.cols()), NN(nodes.size());
244 const bool TARGET_H_OK(target_h > 0.);
245 DoubleMat pos3d, pos3d_, T, T_;
246 DoubleVec H_, metrics1d, metrics1d_, pos1d_;
247 UIntVec indices, indices_, hnodes;
248 size_t n0, n1, n2;
249 double min_h1d(0.), max_h1d(0.);
250 double h0, h1, L01, U01;
251 int ret(0); // The return code.
252
253#ifdef CM2_DBG_FILES
254 // Output of the parameters.
255 std::ofstream dbg_strm("dbg_info.dat");
256 dbg_strm.flags(std::ios::scientific);
257 dbg_strm.precision(16);
258 dbg_strm << "target_h = " << target_h << std::endl;
259 dbg_strm << "force_even = " << int(force_even) << std::endl;
260 dbg_strm << "min_n = " << min_n << std::endl;
261 dbg_strm << "max_n = " << max_n << std::endl;
262 dbg_strm << "max_chordal_error = " << max_chordal_error << std::endl;
263 dbg_strm << "min_h = " << min_h << std::endl;
264 dbg_strm << "chordal_control_type = " << chordal_control_type << std::endl;
265 dbg_strm << "max_gradation = " << max_gradation << std::endl;
266 dbg_strm << "high_order_type = " << high_order_type << std::endl;
267 dbg_strm << "high_order_mode = " << high_order_mode << std::endl;
268#endif
269
270 connectE.clear();
271
272 // Check some input data.
273 if (NN < 2) return -3; // At least two nodes.
274 n0 = vecscal::max_value(nodes);
275 if (n0 >= NP) return -3;
276 if (!pos1d.empty() && (pos1d.size() != NN)) return -4;
277
278 // The metrics of the initial nodes (background-mesh nodes).
279 if (!H.resize(NN, target_h))
280 return -199; // Memory error.
281 meshtools::overwrite_invalid_metrics(H, target_h);
282 if (vecscal::min_value(H) <= 0.) return -5;
283 h0 = H.front();
284 h1 = H.back();
285
286 // Add null Z coordinates to the 3-D points if needed.
287 if (!pos.resize(3, NP, 0.))
288 return -199; // Memory error.
289
290 // Compress the 3-D coordinates of the initial nodes (background-mesh nodes).
291 if (!matmat::push_back(pos, nodes, pos3d))
292 return -199; // Memory error.
293
294 n0 = n1 = NN;
295 indices.resize(n0);
296 for (size_t i = 0; i < NN; ++i) indices[i] = static_cast<unsigned>(i); // iota.
297
298 // Compute the reference coordinates pos1d if missing.
299 if (pos1d.empty())
300 {
301 pos1d.resize(n0, 0.);
302 if (C.get_1D_coordinates(pos3d, pos1d) < 0)
303 return -200; // Internal error.
304 if (!__is_sorted(pos1d.begin(), pos1d.end()))
305 return -3; // U coordinates should be monotonous ascendingly.
306 }
307 else if (!__is_sorted(pos1d.begin(), pos1d.end()))
308 return -4; // U coordinates should be monotonous ascendingly.
309
310 // Compute the tangents for all nodes.
311 // The local bases will be used to transform the 3-D metrics into 1-D metrics.
312 if (!T.resize(3, n0, 0.))
313 return -199;
314 if (C.get_tangents(pos1d, T) < 0)
315 return -200; // Internal error.
316 assert((T.rows() == 3) && (T.cols() == n0));
317
318 // Transform the 3-D metrics into 1-D metrics.
319 metrics1d.resize(n0, 0.);
320 if (__metrics_3d_to_1d(H, T, metrics1d) < 0)
321 return -200; // Internal error.
322
323#ifdef CM2_DBG_FILES
324 // Output of the reference coordinates and local bases.
325 dbg_strm << "Ref coordinates.\n";
326 vecio::write(dbg_strm, pos1d);
327 dbg_strm << std::endl;
328 dbg_strm << "Local bases.\n";
329 matio::transpose_write(dbg_strm, T);
330 dbg_strm << std::endl;
331#endif
332
333 min_n = std::max(min_n, 1u);
334 max_n = std::max(max_n, min_n);
335 U01 = pos1d.back() - pos1d.front();
336
337 // 1-D size upper bounds.
338 max_h1d = U01 / min_n;
339 max_h1d = std::max(max_h1d, min_h1d);
340 min_h1d = std::max(min_h1d, 1E-6*max_h1d);
341#ifdef CM2_DBG_FILES
342 dbg_strm << "min_h1d = " << min_h1d << std::endl;
343 dbg_strm << "max_h1d = " << max_h1d << std::endl << std::endl;
344#endif
345
346 // If missing or forced, compute an adapted background mesh.
347 new_bgm_flag |= (NN == 2);
348 if (new_bgm_flag)
349 {
350 for (unsigned mesh_iter = 0; (mesh_iter <= max_bgm_remeshings) && (ret >= 0); ++mesh_iter)
351 {
352 connectE.clear();
353 meshtools1d::indices_to_connectE2(indices, connectE);
354
355 // Clamp the 1-D metrics.
356 ret = meshtools::bound_metrics(metrics1d, min_h1d, max_h1d);
357
358 // Reduce some metrics to limit size gradation.
359 if (ret >= 0)
360 {
361 DoubleMat pos1dm(2, pos1d.size(), 0.);
362 vecvec::copy_n(pos1d.begin(), pos1dm.row_begin(0), pos1d.size());
363 ret = meshtools::bound_metric_gradations(pos1dm, connectE, CM2_EDGE2, // Requires pos1dm as a matrix.
364 UIntVec(), true, -1,
365 /*max_gradation=>*/ 0.15,
366 metrics1d);
367 }
368
369 // (Re)generate the background mesh.
370 if (ret >= 0)
371 {
372 std::swap(indices, indices_);
373 indices.clear();
374 ret = meshtools1d::subdivide(pos1d, metrics1d, indices_,
375 /*force_up=>*/ true,
376 /*force_even=>*/ false,
377 /*min_n=>*/ static_cast<unsigned>(indices_.size() + 2),
378 /*max_n=>*/ UINT_MAX,
379 indices);
380 assert((indices.front() == indices_.front()) && (indices.front() == 0));
381 assert((indices.back() == indices_.back()) && (indices.back() == n0 - 1));
382 }
383 if (ret < 0)
384 break;
385
386 // Eliminate the old BGM nodes (keep only the new indices).
387 std::copy(pos1d.begin() + n1, pos1d.end(), pos1d.begin() + n0);
388 pos1d.pop_back(n1 - n0);
389 std::copy(metrics1d.begin() + n1, metrics1d.end(), metrics1d.begin() + n0);
390 metrics1d.pop_back(n1 - n0);
391 vecscal::subtract_n(static_cast<unsigned>(n1 - n0), indices.begin() + 1, indices.size() - 2); // Keep first and last indices (0 and n0-1).
392 n1 = pos1d.size();
393
394 if (!H.resize(n1) || !T.resize(3, n1))
395 return -199; // Memory error.
396 pos1d_ = pos1d.sub_vector(n0, n1 - n0); // New nodes of the BGM.
397 metrics1d_ = metrics1d.sub_vector(n0, n1 - n0); //
398 H_ = H.sub_vector(n0, n1 - n0); //
399
400 // Compute the local bases at the new nodes.
401 T_ = T.sub_matrix(0, n0, 3, n1 - n0);
402 if (C.get_tangents(pos1d_, T_) < 0)
403 return -200; // Internal error.
404
405 // Set 1-D metric on the new nodes.
406 ret = meshtools1d::interpolate_metrics(pos1d.sub_vector(0, n0), H.sub_vector(0, n0),
407 pos1d_, H_);
408 if (ret >= 0)
409 ret = __metrics_3d_to_1d(H_, T_, metrics1d_);
410
411 // Take the gradations of the tangents into account.
412 if (ret >= 0)
413 {
414 L01 = __approximative_length(indices, pos1d, T);
415 ret = __tangent_control(U01*U01/L01, T, min_h1d, metrics1d);
416 }
417
418 // Take the curvatures into account.
419 if (ret >= 0)
420 ret = __chordal_control(C, pos1d, indices, T,
421 /*max_chordal_error=>*/ -0.01, // 1%
422 /*chordal_control_type=>*/ 2, // Use exact curvatures if possible.
423 min_h,
424 metrics1d,
425 /*flag1D=>*/ true);
426 if (ret < 0)
427 break;
428
429 // Compute Qh_max.
430 const double QH_MAX(1.10); // Stop remeshing when Qh_max is below this value.
431 double qh, Qh_max(0.);
432 double ui, uj(pos1d.front());
433 double hi, hj(metrics1d.front());
434 size_t Nj(indices.front());
435 for (size_t j = 1; j < indices.size(); ++j)
436 {
437 ui = uj; hi = hj;
438 Nj = indices[j]; uj = pos1d[Nj]; hj = metrics1d[Nj];
439 assert(uj > ui);
440 meshtools::edge_quality(uj-ui, hi, hj, qh); // Edge quality in the 1-D coordinate.
441 Qh_max = std::max(Qh_max, qh);
442 }
443 if (Qh_max < QH_MAX)
444 break;
445 } // mesh_iter.
446 } // if (new_bgm_flag).
447
448 connectE.clear();
449 if (ret < 0)
450 return ((ret != -199) ? ret : -200); // Memory or internal error.
451
452 if (!pos3d.resize(3, n1, 0.))
453 return -199; // Memory error.
454 meshtools1d::indices_to_connectE2(indices, connectE);
455
456 pos1d_ = pos1d.sub_vector(n0, n1 - n0);
457 pos3d_ = pos3d.sub_matrix(0, n0, 3, n1 - n0);
458 H_ = H.sub_vector(n0, n1 - n0);
459
460 // Set the 3-D metrics on the nodes of the BGM.
461 if ((NN == 2) && TARGET_H_OK) // Uniform H on inside nodes.
462 H_ = target_h;
463 else // Interpolate the metrics of the hard nodes onto the nodes of the background mesh (nodes n0 -> n1).
464 ret = meshtools1d::interpolate_metrics(pos1d.sub_vector(0, n0), H.sub_vector(0, n0), pos1d_, H_);
465
466 // Map the new 1-D nodes onto the 3-D curve.
467 if (C.get_3D_coordinates(pos1d_, pos3d_) < 0)
468 return -200; // Internal error.
469
470 // Adapt the inner 3-D metrics (up and down) to respect max_gradation (H[0] and H[n0-1] are fixed).
471 hnodes.clear();
472 hnodes.push_back(0, static_cast<unsigned>(n0 - 1)); // Fix first and last values in H.
473 if (ret >= 0)
474 ret = meshtools::bound_metric_gradations(pos3d,
475 connectE, CM2_EDGE2,
476 hnodes, true, 0, max_gradation,
477 H);
478
479 assert((H[0] == h0) && (H[n0-1] == h1)); // No change so far on H[0] and H[n0-1].
480
481 // Bound the 3-D metrics (down only) with curvature metrics.
482 if (ret >= 0)
483 ret = __chordal_control(C, pos1d, indices, T,
484 max_chordal_error, chordal_control_type, min_h,
485 H, /*flag1D=>*/ false);
486
487 // Adapt again the 3-D metrics (down only) to respect max_gradation.
488 if (!update_metrics_flag) // Restore H to initial values.
489 {
490 H[0] = h0;
491 H[n0 - 1] = h1;
492 }
493 else
494 hnodes.clear();
495 if (ret >= 0)
496 ret = meshtools::bound_metric_gradations(pos3d,
497 connectE, CM2_EDGE2,
498 hnodes, true, -1, max_gradation,
499 H);
500
501 // Transform the 3-D metrics into 1-D metrics.
502 if (ret >= 0)
503 ret = __metrics_3d_to_1d(H, T, metrics1d);
504
505 // Clamp the 1-D metrics.
506 if (ret >= 0)
507 ret = meshtools::bound_metrics(metrics1d, min_h1d, max_h1d);
508
509#ifdef CM2_DBG_FILES
510 // Output metrics1d after curvature bounding.
511 dbg_strm << "metrics1d1.\n";
512 vecio::write(dbg_strm, metrics1d);
513 dbg_strm << std::endl;
514#endif
515
516 if (ret < 0)
517 return -200; // Internal error.
518
519 // Now, we can do the 1-D mesh.
520 std::swap(indices, indices_);
521 indices.clear();
522 ret = meshtools1d::subdivide(pos1d, metrics1d, indices_,
523 /*force_up=>*/ false,
524 force_even, min_n, max_n,
525 indices);
526 if (ret < 0)
527 return -200; // Internal error.
528
529 n2 = pos1d.size();
530
531 // New 3-D metrics.
532 {
533 DoubleVec pos1d__, H__;
534
535 // H at new nodes.
536 if (!H.resize(n2, 0.))
537 return -199; // Memory error.
538 if (new_bgm_flag)
539 {
540 pos1d__.push_back(pos1d[0]);
541 std::copy(pos1d.data() + n0, pos1d.data() + n1, std::back_inserter(pos1d__));
542 pos1d__.push_back(pos1d[n0-1]);
543 H__.push_back(h0);
544 std::copy(H.data() + n0, H.data() + n1, std::back_inserter(H__));
545 H__.push_back(h1);
546 }
547 else
548 {
549 pos1d__ = pos1d.sub_vector(0, n0);
550 H__ = H.sub_vector(0, n0);
551 }
552 pos1d_ = pos1d.sub_vector(n1, n2 - n1);
553 H_ = H.sub_vector(n1, n2 - n1);
554 ret = meshtools1d::interpolate_metrics(pos1d__, H__, pos1d_, H_);
555 if (ret < 0)
556 return -200; // Internal error.
557 }
558
559 // Eliminate the new BGM nodes (between n0 and n1).
560 if (new_bgm_flag)
561 {
562 std::copy(pos1d.begin() + n1, pos1d.end(), pos1d.begin() + n0);
563 pos1d.pop_back(n1 - n0);
564 std::copy(metrics1d.begin() + n1, metrics1d.end(), metrics1d.begin() + n0);
565 metrics1d.pop_back(n1 - n0);
566 std::copy(H.begin() + n1, H.end(), H.begin() + n0);
567 H.pop_back(n1 - n0);
568 vecscal::subtract_n(static_cast<unsigned>(n1 - n0), indices.begin() + 1, indices.size() - 2); // Keep first and last indices (0 and n0-1).
569 }
570
571 n1 = pos1d.size();
572
573 // Map the new 1-D nodes onto the 3-D curve.
574 if (!pos3d.resize(3, n1, 0.))
575 return -199; // Memory error.
576 pos1d_ = pos1d.sub_vector(n0, n1 - n0);
577 pos3d_ = pos3d.sub_matrix(0, n0, 3, n1 - n0);
578 if (C.get_3D_coordinates(pos1d_, pos3d_) < 0)
579 return -200; // Internal error.
580
581 connectE.clear();
582 meshtools1d::indices_to_connectE2(indices, connectE);
583
584#ifdef CM2_DBG_FILES
585 // Output of the 3-D anisotropic mesh.
586 meshtools::vizir_output("mesh3d.mesh", pos3d, connectE, CM2_EDGE2);
587#endif
588
589 // Convert to high-degree elements if required.
590 if (high_order_type > 0)
591 {
592 // Generate HON interpolated between the 3-D nodes (straight elements).
593 n1 = pos3d.cols();
594 meshtools1d::convert_into_high_order(pos3d, connectE, /*Ne=>*/ high_order_type);
595 n2 = pos3d.cols();
596
597 // Generate the U coordinates of HON (interpolated between 1-D nodes).
598 {
599 DoubleMat pos1dm(1, n1, 1, pos1d); // pos1d as a 1-row matrix (share data).
600 connectE.resize(2, connectE.cols());
601 meshtools1d::convert_into_high_order(pos1dm, connectE, /*Ne=>*/ high_order_type);
602 pos1d = pos1dm.as_vector();
603 assert(pos1d.size() == pos3d.cols());
604 pos1d_ = pos1d.sub_vector(n1, n2 - n1); // View to the new high-order nodes.
605 pos3d_ = pos3d.sub_matrix(0, n1, 3, n2 - n1); // View to the new high-order nodes.
606 }
607
608 // Compute the U coordinates of 3-D interpolated HON (U of interpolated XYZ-HON).
609 // This may be time consuming especially if Curve::get_1D_coordinates discard the initial U values computed above.
610 if (high_order_mode >= 2)
611 {
612 if (C.get_1D_coordinates(pos3d_, pos1d_) < 0) // Upon entry pos1d_ are already good estimates (interpolated U-HON).
613 return -200;
614 }
615
616 // Project the new high-order nodes onto the curve.
617 if (high_order_mode >= 1)
618 {
619 if (C.get_3D_coordinates(pos1d_, pos3d_) < 0)
620 return -200;
621 }
622
623 // Null H at high-order nodes.
624 H.push_back_n(0., n2 - n1);
625 }
626
627 if (ret < 0)
628 return -200; // Internal error.
629
630 // Back to uncompressed node IDs.
631 n2 = pos3d.cols();
632 if (!pos.push_back(pos3d.sub_matrix(0, n0, 3, n2 - n0)))
633 return -199; // Memory error.
634 matscal::add(static_cast<unsigned>(NP - n0), connectE);
635 connectE(0, 0) = nodes.front();
636 connectE(1, connectE.cols() - 1) = nodes.back();
637
638 assert(pos1d.size() == H.size());
639
640 return 0; // OK.
641}

664template <class Curve>
665int
666mesh_curve_param
667 (const Curve& C,
668 DoubleMat& pos, const UIntVec& nodes, DoubleVec& pos1d, DoubleMat& metrics, UIntMat& connectE,
669 double target_h, bool force_even, unsigned min_n, unsigned max_n,
670 double max_chordal_error, double min_h, unsigned chordal_control_type, double max_gradation,
671 unsigned high_order_type = 0, unsigned high_order_mode = 1,
672 bool new_bgm_flag = true, unsigned max_bgm_remeshings = 4, bool update_metrics_flag = true)
673{
674 const size_t NP(pos.cols()), NN(nodes.size());
675 const bool TARGET_H_OK(target_h > 0.);
676 DoubleMat T;
677 DoubleVec H;
678 DoubleVec3 Tu, Cu;
679 DoubleSym3 target_M, Mi;
680 double c2;
681 size_t n1;
682 int ret(0);
683
684 // Case metrics is indeed isotropic.
685 if ((metrics.rows() == 1) && (metrics.ld() == 1))
686 {
687 H = metrics.as_vector();
688 ret = mesh_curve_param(C, pos, nodes, pos1d, H, connectE,
689 target_h, force_even, min_n, max_n,
690 max_chordal_error, min_h, chordal_control_type, max_gradation,
691 high_order_type, high_order_mode,
692 new_bgm_flag, max_bgm_remeshings, update_metrics_flag);
693 metrics.resize(1, H.size());
694 return ret;
695 }
696
697 connectE.clear();
698
699 // Check some input data.
700 if (NN < 2) return -3; // At least two nodes.
701 if (vecscal::max_value(nodes) >= NP) return -3;
702 if (!pos1d.empty() && (pos1d.size() != NN)) return -4;
703 if (!metrics.empty() && (metrics.rows() != 6)) return -5;
704
705 // Metrics.
706 if (!metrics.resize(6, NN, 0.))
707 return -199; // Memory error.
708 meshtools::metric_aniso(target_h, target_M);
709 meshtools::overwrite_invalid_metrics(metrics, target_M);
710 if (!meshtools::are_valid_metrics(metrics)) return -5;
711
712 // Compute curvilinear coordinates of initial nodes if missing.
713 if (pos1d.empty())
714 {
715 DoubleMat pos3d;
716 matmat::push_back(pos, nodes, pos3d); // Compress coordinates.
717 pos1d.resize(NN, 0.);
718 if (C.get_1D_coordinates(pos3d, pos1d) < 0)
719 return -200; // Internal error.
720 }
721
722 // Compute the mesh sizes in the tangent directions (scalars H).
723 if (!T.resize(3, NN) || !H.resize(NN, 0.))
724 return -199; // Memory error.
725 if (C.get_tangents(pos1d, T) < 0)
726 return -200; // Internal error.
727 __metrics_3d_aniso_to_iso(metrics, T, H);
728
729 // Mesh the curve.
730 ret = mesh_curve_param(C, pos, nodes, pos1d, H, connectE,
731 target_h, force_even, min_n, max_n,
732 max_chordal_error, min_h, chordal_control_type, max_gradation,
733 high_order_type, high_order_mode,
734 new_bgm_flag, max_bgm_remeshings, update_metrics_flag);
735 if (ret < 0)
736 return ret;
737
738 // Update 3-D metrics.
739 metrics.resize(6, H.size(), 0.);
740
741 if (update_metrics_flag)
742 {
743 // Update metric of first node.
744 vecvec::copy_n(metrics.col_begin(0), Mi.data(), 6);
745 C.get_derivs(pos1d[0], Tu, Cu, c2);
746 __reduce_3d_metric(Mi, H[0], Tu, Cu, c2);
747 vecvec::copy_n(Mi.data(), metrics.col_begin(0), 6);
748
749 // Update metric of last node.
750 vecvec::copy_n(metrics.col_begin(NN - 1), Mi.data(), 6);
751 C.get_derivs(pos1d[NN - 1], Tu, Cu, c2);
752 __reduce_3d_metric(Mi, H[NN - 1], Tu, Cu, c2);
753 vecvec::copy_n(Mi.data(), metrics.col_begin(NN - 1), 6);
754 }
755
756 // Metrics on new nodes.
757 n1 = pos1d.size() - connectE.cols() * high_order_type; // Nbr of linear nodes.
758 if ((NN == 2) && TARGET_H_OK) // Uniform H on inside nodes.
759 {
760 for (size_t i = NN; i < n1; ++i)
761 vecvec::copy_n(target_M.data(), metrics.col_begin(i), 6);
762 }
763 else
764 {
765 const DoubleVec pos1d__(pos1d.sub_vector(0, NN));
766 const DoubleVec pos1d_(pos1d.sub_vector(NN, n1 - NN));
767 const DoubleMat metrics__(metrics.sub_matrix(0, 0, 6, NN));
768 DoubleMat metrics_(metrics.sub_matrix(0, NN, 6, n1 - NN));
769
770 ret = meshtools1d::interpolate_metrics(pos1d__, metrics__, pos1d_, metrics_);
771 if (ret < 0)
772 return -200; // Internal error.
773 }
774 for (size_t i = NN; i < n1; ++i)
775 {
776 vecvec::copy_n(metrics.col_begin(i), Mi.data(), 6);
777 C.get_derivs(pos1d[i], Tu, Cu, c2);
778 __reduce_3d_metric(Mi, H[i], Tu, Cu, c2);
779 vecvec::copy_n(Mi.data(), metrics.col_begin(i), 6);
780 }
781
782 assert(metrics.cols() == pos1d.size());
783
784 return 0;
785}

793inline double
794__approximative_length (const UIntVec& indices, const DoubleVec& pos1d, const DoubleMat& T)
795{
796 const size_t NN(indices.size());
797 double L(0.), ui, uj, ti, tj;
798 UIntVec::const_iterator pi(indices.begin());
799 size_t Nj;
800
801 if (indices.empty())
802 return 0.;
803
804 assert(T.rows() == 3);
805
806 Nj = *pi++;
807 uj = pos1d[Nj];
808 tj = vecscal::two_norm(T.seg(Nj));
809 for (size_t j = 1; j < NN; ++j, ++pi)
810 {
811 ui = uj; ti = tj;
812 Nj = *pi;
813 uj = pos1d[Nj];
814 assert(uj > ui);
815 tj = vecscal::two_norm(T.seg(Nj));
816 L += (ti + tj) * (uj - ui);
817 }
818
819 return 0.5 * L;
820}

912inline void
913__reduce_3d_metric (DoubleSym3& M, double h, const DoubleVec3& T, const DoubleVec3& C, double c2)
914{
915 const double h2(h * h);
916 const double t2(vecscal::sqr_two_norm(T));
917 DoubleSym3 M_(0.);
918 double d;
919
920 if (h2 * t2 == 0.) // Error.
921 return;
922
923// assert(meshtools::is_valid_metric(M));
924
925 // Set M_ to h in the tangent direction.
926 d = 1. / (t2 * h2);
927 matvec::rank1(d, T, M_);
928
929 // Set M_ to h in the normal direction if curvature.
930 if (c2 * h2 > CM2_EPSILON2)
931 {
932 DoubleVec3 Vn(0.), Vb(0.);
933 vecvec::cross(T, C, Vb);
934 vecvec::cross(Vb, T, Vn);
935 d = vecscal::sqr_two_norm(Vn);
936 assert(d > 0.);
937 d = 1. / (d * h2);
938 matvec::rank1(d, Vn, M_);
939 }
940
941 // Reduce M.
942 meshtools::minimum_metric(M, M_, M);
943
944// assert(meshtools::is_valid_metric(M));
945}

948inline int
949__tangent_control (double s, const DoubleMat& T, double min_h1d, DoubleVec& M1D)
950{
951 const size_t NN(T.cols()), LD(T.ld());
952 DoubleMat::const_iterator pl(T.begin());
953 DoubleVec::iterator p1d(M1D.begin());
954 double h;
955
956 if (NN == 0)
957 return 0;
958
959 assert(T.rows() == 3);
960 assert(M1D.size() == NN);
961
962 for (size_t j = 0; j < NN; ++j, ++p1d, pl += LD)
963 {
964 const double t = vecscal::sqr_two_norm3(pl);
965 h = s * ::sqrt(t);
966 h = std::max(h, min_h1d);
967 *p1d = std::min(*p1d, h);
968 }
969
970 return 0; // OK.
971}

974template <class Curve>
975int
976__chordal_control
977 (const Curve& C,
978 const DoubleVec& pos1d, const UIntVec& bgm_nodes, const DoubleMat& tangents,
979 double max_chordal_error, unsigned chordal_control_type, double min_h3d,
980 DoubleVec& metrics, bool flag1D)
981{
982 const size_t NODS(pos1d.size());
983 UIntMat connectE;
984 DoubleVec Curvs;
985 DoubleVec::iterator p1, pc;
986 DoubleMat::const_iterator pL;
987 double t2(1.), c, eps, h;
988 int ret(0);
989
990 // Fast return.
991 if ((chordal_control_type == 0) || (max_chordal_error == 0.) || (max_chordal_error == DBL_MAX)
992 || (max_chordal_error < -1.))
993 return 0;
994
995 assert(tangents.cols() == NODS);
996 assert(tangents.rows() == 3);
997 assert(tangents.ld() == 3);
998 assert(metrics.size() == NODS);
999
1000 // Compute the curvature at each node.
1001 if (chordal_control_type == 1)
1002 {
1003 meshtools1d::indices_to_connectE2(bgm_nodes, connectE);
1004 ret = meshtools1d::parametric_curvatures(pos1d, connectE, tangents, Curvs); // Approximative curvatures.
1005 }
1006 else
1007 {
1008 if (!Curvs.resize(NODS))
1009 return -199; // Memory error.
1010 ret = C.get_curvatures(pos1d, Curvs); // Exact curvatures (user-supplied).
1011
1012 // If C.get_curvatures is bogus, fall-back to approximative curvatures.
1013 if (ret < 0)
1014 {
1015 meshtools1d::indices_to_connectE2(bgm_nodes, connectE);
1016 ret = meshtools1d::parametric_curvatures(pos1d, connectE, tangents, Curvs);
1017 }
1018 }
1019 if (ret < 0)
1020 return ret;
1021
1022 // Bound the metrics with the curvature metrics.
1023 assert(Curvs.size() == NODS);
1024 pc = Curvs.begin();
1025 pL = tangents.begin();
1026 p1 = metrics.begin();
1027 for (size_t i = 0; i < NODS; ++i, ++pc, ++p1, pL += 3)
1028 {
1029 c = *pc;
1030 if (c <= 0.)
1031 continue;
1032 if (c * min_h3d > 2.) // Consider very high curvatures as pathological and discard.
1033 continue;
1034 if (flag1D)
1035 t2 = vecscal::sqr_two_norm3(pL);
1036 if (t2 == 0.)
1037 continue;
1038 if (max_chordal_error > 0.) // Absolute chordal error.
1039 eps = max_chordal_error * c;
1040 else // Relative chordal error (relative to curvatures).
1041 eps = -max_chordal_error;
1042 eps = std::max(eps, 1E-6); // Clamp eps between 1E-6 and 1.
1043 eps = std::min(eps, 1.); //
1044 h = (2. * ::sqrt(eps*(2.-eps)/t2)) / c;
1045 h = std::max(h, min_h3d);
1046 *p1 = std::min(*p1, h);
1047 }
1048
1049 return ret;
1050}
