#include "stdafx.h"
#include "tests_tmsh2da.h"
/*
This example shows how to use the background mesh.
Here, the background mesh covers the whole domain and is supporting a varying metric.
*/
void
TMSH2DA56::run
(double R, double h0, double h1, double sig, unsigned func_ID, const char* mesh_type)
{
const cm2::element_type FE_type = the_fe_type(mesh_type);
const cm2::DoubleVec2 P0(R, 0.);
const cm2::DoubleVec2 CR(0.);
const double sig_inv = M_SQRT1_2 / sig;
const unsigned N_BGM = unsigned(2.*M_PI*R / h1); // The discretization for the background mesh.
cm2::DoubleMat pos;
cm2::DoubleMat metrics, metrics0, new_metrics;
cm2::UIntVec indices0, indices;
cm2::UIntMat connectE, connectM, BGM;
cm2::DoubleMat2x2 B;
cm2::DoubleSym2 Mi, Mj, Mk;
unsigned n;
double x, y, w, h;
cm2::DoubleVec new_U;
cm2::triamesh_iso::mesher aux_mesher;
print_header("TMSH2DA56 - DISK WITH BACKGROUND MESH", mesh_type);
cm2::matvec::push_back(CR, pos);
cm2::matvec::push_back(P0, pos);
// THE BACKGROUND MESH.
cm2::meshtools1d::extrude_rotate(pos, P0, CR, 2.*M_PI, N_BGM, indices);
indices.back() = indices.front();
cm2::meshtools1d::indices_to_connectE2(indices, connectE);
cm2::triamesh_iso::mesher::data_type BGMdata(pos, connectE);
BGMdata.isolated_nodes.push_back(0); // This accelerates the generation of the mesh.
aux_mesher.settings.optim_level = 0;
aux_mesher.run(BGMdata);
BGMdata.extract(pos, BGM);
// MESH THE GEOMETRIC SUPPORT OF THE EXTERNAL CONTOUR (CIRCLE).
cm2::meshtools1d::extrude_rotate(pos, 1, CR, 2*M_PI, R/100., R/100., false, indices0);
indices0.back() = indices0.front();
// THE METRICS ON THE BACKGROUND MESH AND ON THE CIRCLE.
indices.clear();
cm2::meshtools::unique_indices(indices, BGM);
std::copy(indices0.begin(), indices0.end(), std::back_inserter(indices));
metrics.resize(3, pos.cols(), 0.);
for (size_t i = 0; i < indices.size(); ++i)
{
n = indices[i];
x = pos(0,n);
y = pos(1,n);
switch (func_ID)
{
case 0 : w = ::fabs(y) * sig_inv; // Horizontal line y = 0
Mk(0, 0) = 1. / (h0*h0);
Mk(0, 1) = 0.;
h = (1./h0) + (1./h1) * ::exp(-w*w);
Mk(1, 1) = h * h;
break;
case 1 : w = ::fabs(y - x - R/4.) * sig_inv; // Diagonal line y = x + R/4
Mk(0, 0) = 1. / (h0*h0);
Mk(0, 1) = 0.;
h = (1./h0) + (1./h1) * ::exp(-w*w);
Mk(1, 1) = h * h;
B(0, 0) = +M_SQRT1_2; B(1, 0) = M_SQRT1_2;
B(0, 1) = -M_SQRT1_2; B(1, 1) = M_SQRT1_2;
vecscal::copy(0., Mi);
matmat::B_A_Bt (1., Mk, B, Mi);
vecvec::copy(Mi, Mk);
break;
case 2 : w = ::fabs(y - x) * sig_inv; // Diagonal line y = x
Mk(0, 0) = 1. / (h0*h0);
Mk(0, 1) = 0.;
h = (1./h0) + (1./h1) * ::exp(-w*w);
Mk(1, 1) = h * h;
B(0, 0) = +M_SQRT1_2; B(1, 0) = M_SQRT1_2;
B(0, 1) = -M_SQRT1_2; B(1, 1) = M_SQRT1_2;
vecscal::copy(0., Mi);
matmat::B_A_Bt (1., Mk, B, Mi);
w = ::fabs(y + x) * sig_inv; // Diagonal line y = -x
Mk(0, 0) = 1. / (h0*h0);
Mk(0, 1) = 0.;
h = (1./h0) + (1./h1) * ::exp(-w*w);
Mk(1, 1) = h * h;
B(0, 0) = +M_SQRT1_2; B(1, 0) = -M_SQRT1_2;
B(0, 1) = +M_SQRT1_2; B(1, 1) = +M_SQRT1_2;
vecscal::copy(0., Mj);
matmat::B_A_Bt (1., Mk, B, Mj);
meshtools::minimum_metric (Mi, Mj, Mk);
break;
case 3 : w = (::fabs(y + 2.*x - R/2) /::sqrt(5.)) * sig_inv; // 3 lines.
Mk(0, 0) = 1. / (h0*h0);
Mk(0, 1) = 0.;
h = (1./h0) + (1./h1) * ::exp(-w*w);
Mk(1, 1) = h * h;
B(0, 0) = +1./sqrt(5.); B(1, 0) = -2./sqrt(5.);
B(0, 1) = +2./sqrt(5.); B(1, 1) = +1./sqrt(5.);
vecscal::copy(0., Mi);
matmat::B_A_Bt(1., Mk, B, Mi);
w = ::fabs(y - x - R/2) * sig_inv;
Mk(0, 0) = 1. / (h0*h0);
Mk(0, 1) = 0.;
h = (1./h0) + (1./h1) * ::exp(-w*w);
Mk(1, 1) = h * h;
B(0, 0) = +M_SQRT1_2; B(1, 0) = M_SQRT1_2;
B(0, 1) = -M_SQRT1_2; B(1, 1) = M_SQRT1_2;
vecscal::copy(0., Mj);
matmat::B_A_Bt(1., Mk, B, Mj);
meshtools::minimum_metric(Mi, Mj, Mk);
vecvec::copy(Mk, Mi);
w = ::fabs(x + R/2) * sig_inv;
Mk(0, 0) = 1. / (h0*h0);
Mk(0, 1) = 0.;
h = (1./h0) + (1./h1) * ::exp(-w*w);
Mk(1, 1) = h * h;
B(0, 0) = 0.; B(1, 0) = 1.;
B(0, 1) = -1.; B(1, 1) = 0.;
vecscal::copy(0., Mj);
matmat::B_A_Bt(1., Mk, B, Mj);
meshtools::minimum_metric(Mi, Mj, Mk);
break;
default : break;
}
metrics(0, n) = Mk(0, 0);
metrics(1, n) = Mk(0, 1);
metrics(2, n) = Mk(1, 1);
}
// Pickup the sizes along the circle.
metrics0.clear();
matmat::push_back(metrics, indices0, metrics0);
// MESH THE EXTERNAL CONTOUR.
indices.clear();
connectE.clear();
cm2::meshtools1d::mesh_line (pos, indices0, metrics0,
true, 1, UINT_MAX, 0., 0.,
indices, new_U, new_metrics);
cm2::meshtools1d::indices_to_connectE2(indices, connectE);
// THE 2D MESH.
if (FE_type == CM2_FACET3)
{
cm2::triamesh_aniso::mesher the_mesher;
cm2::triamesh_aniso::mesher::data_type data(pos, connectE);
data.background_mesh = BGM;
data.metrics = metrics;
the_mesher.settings.reset();
the_mesher.run(data);
data.extract(pos, connectM);
data.print_info(&display_hdl);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
else
{
cm2::quadmesh_aniso::mesher the_mesher;
cm2::quadmesh_aniso::mesher::data_type data(pos, connectE);
the_mesher.settings.all_quad_flag = (FE_type == CM2_FACEQ4);
data.background_mesh = BGM;
data.metrics = metrics;
the_mesher.run(data);
data.extract(pos, connectM);
data.print_info(&display_hdl);
print_footer(data.nods, data.nefs, data.total_time, data.speed);
}
// OUTPUT FOR VISUALISATION.
char filename[64];
::strcpy(filename, mesh_type);
::strcat(filename, ".mesh");
cm2::meshtools::vizir_output(filename, pos, connectM, FE_type);
}
