Example tmsh2db02.cpp (pure quadrangle)
#include "stdafx.h"
#include "tests_tmsh2db.h"
/*
Non-manifold square 1.
This test creates a 2D-layer in a non-manifold square for several cases of discretization and several depth and growth rate values.
\param[in] depth total depth of the layers.
\param[in] L Number of layers.
\param[in] rho Growth ratio ((n+1)th layer depth over nth layer depth).
*/
void
TMSH2DB02::run (double d, unsigned L, double rho, const char* mesh_type, bool fill_in)
{
const cm2::element_type FE_type = CM2_FACE_MIX;
cm2::DoubleMat pos;
unsigned A, B, C, D, E, F, G;
cm2::UIntVec indices;
cm2::UIntMat connectB, connectC, connectLN;
cm2::DoubleVec depth;
print_header("TMSH2DB02 - NON MANIFOLD SQUARE 1", mesh_type);
// VERTICES.
pos.push_back(DoubleVec2(0., 0.)); A = pos.cols()-1;
pos.push_back(DoubleVec2(5., 0.)); B = pos.cols()-1;
pos.push_back(DoubleVec2(5., 1.)); C = pos.cols()-1;
pos.push_back(DoubleVec2(5., 2.)); D = pos.cols()-1;
pos.push_back(DoubleVec2(10., 0.)); E = pos.cols()-1;
pos.push_back(DoubleVec2(10., 10.)); F = pos.cols()-1;
pos.push_back(DoubleVec2(0., 10.)); G = pos.cols()-1;
// CONTOUR MESH.
connectB.push_back(UIntVec2(A, B));
connectB.push_back(UIntVec2(B, C));
connectB.push_back(UIntVec2(C, D));
connectB.push_back(UIntVec2(D, C));
connectB.push_back(UIntVec2(C, B));
connectB.push_back(UIntVec2(B, E));
connectB.push_back(UIntVec2(E, F));
connectB.push_back(UIntVec2(F, G));
connectB.push_back(UIntVec2(G, A));
depth.resize(connectB.cols(), d);
// LAYER 2D MESH.
cm2::layers_2d::mesher l2dmesher;
cm2::layers_2d::mesher::data_type data(pos, connectB, depth);
l2dmesher.settings.rho = rho;
l2dmesher.settings.nlayers = L;
if (is_T(mesh_type))
l2dmesher.settings.min_Q4_quality = 2.0;
l2dmesher.run(data);
data.print_info(&display_hdl);
print_footer(data.nods, data.nefs, data.total_time, data.speed, L, d);
// FILL THE INNER DOMAIN WITH TRIANGLES.
if (fill_in)
{
cm2::triamesh_iso::mesher mesher_inside;
fill_inside(mesher_inside, data);
}
// OUTPUT FOR VISUALISATION.
cm2::meshtools::vizir_output("connectB0.mesh", data.pos, data.connectB0, CM2_EDGE2);
cm2::meshtools::vizir_output("connectB1.mesh", data.pos, data.connectB1, CM2_EDGE2);
cm2::meshtools::vizir_output("connectM.mesh", data.pos, data.connectM, FE_type);
}
