52template <class Vec, class Mat>
53bool
54push_back (const Vec& V, Mat& A)
55{
56 // Fast return.
57 if (V.empty())
58 return true;
59
60 const size_t R(V.size()), C(A.cols());
61
62 if (!A.empty() && (A.rows() != R))
63 return false; // Dimension error.
64
65 if (C == A.cols_capacity())
66 A.reserve(R, std::max(size_t(4), 2 * C)); // Double the capacity of the matrix.
67
68 if (!A.resize(R, C + 1))
69 return false; // Memory error.
70
71 vecvec::copy_n(V.begin(), A.col_begin(C), R);
72 return true; // OK.
73}

89template <class Vec, class Mat>
90bool
91insert (const Vec& V, size_t j, Mat& A)
92{
93 if ((j > A.cols()) || !A.push_back(V))
94 return false; // Index error or memory error.
95
96 std::rotate(A.col_begin(j), A.end() - A.ld(), A.end());
97 return true; // OK.
98}

110template <class Mat>
111void
112remove (size_t j, Mat& A)
113{
114 const size_t C(A.cols()), LD(A.ld());
115 typename Mat::iterator pv(A.col_begin(j));
116
117 std::rotate(pv, pv + LD, pv + (C - j) * LD);
118 A.pop_back();
119}

353// MULT //
354// Non-symmetric matrix, column-oriented, full, non-blocked.
355template <class T, class Matrix, class IterX, class IterY>
356void
357__mult (T scale, const Matrix& A,
358 cm2::non_symmetric_tag, cm2::col_oriented_tag, cm2::full_tag, cm2::non_blocked_tag,
359 IterX x, IterY y, int /*nthreads*/)
360{
361 const typename Matrix::const_segment_type Aend(A.seg_end());
362 typename Matrix::const_segment_type A_j;
363
364 // AXPY algorithm.
365 for (A_j = A.seg_begin(); A_j < Aend; ++A_j)
366 {
367 const size_t j(A_j.index()), ibeg(A_j.index_begin()), hj(A_j.size());
368 vecvec::axpy_n(x[j] * scale, A_j.begin(), y+ibeg, hj);
369 }
370}

372// Non-symmetric matrix, row-oriented, full, non-blocked.
373template <class T, class Matrix, class IterX, class IterY>
374void
375__mult (T scale, const Matrix& A,
376 cm2::non_symmetric_tag, cm2::row_oriented_tag, cm2::full_tag, cm2::non_blocked_tag,
377 IterX x, IterY y, int DUM_OMP(nthreads))
378{
379 const size_t N(A.segments());
380 const typename Matrix::const_segment_type A0(A.seg_begin());
381#ifdef _OPENMP
382 const bool parallel_cond = (nthreads > 1) && (N > 1) && (A.entries() > CM2_OMP_MIN_LOOP);
383
384 // Dot product algorithm.
385 if (parallel_cond)
386 {
387 typename Matrix::const_segment_type Ak;
388 #pragma omp parallel for private(Ak) schedule(static) num_threads(nthreads)
389 for (int k = 0; k < int(N); ++k) // Parallel loop over A-segments.
390 {
391 Ak = A0; Ak += k;
392 const size_t i(Ak.index()), jbeg(Ak.index_begin()), hi(Ak.size());
393 y[i] += scale * vecvec::dot_n(Ak.begin(), x + jbeg, hi);
394 }
395 }
396 else
397#endif
398 {
399 typename Matrix::const_segment_type Ak(A0);
400 for (size_t k = 0; k < N; ++k, ++Ak)
401 {
402 const size_t i(Ak.index()), jbeg(Ak.index_begin()), hi(Ak.size());
403 y[i] += scale * vecvec::dot_n(Ak.begin(), x + jbeg, hi);
404 }
405 }
406}

408// Non-symmetric matrix, col-oriented, sparse, non-blocked.
409template <class T, class Matrix, class IterX, class IterY>
410void
411__mult (T scale, const Matrix& A,
412 cm2::non_symmetric_tag, cm2::col_oriented_tag, cm2::sparse_tag, cm2::non_blocked_tag,
413 IterX x, IterY y, int /*nthreads*/)
414{
415 const typename Matrix::const_segment_type Aend(A.seg_end());
416 typename Matrix::const_segment_type Ak;
417 typename Matrix::const_segment_type::iterator Aij;
418 size_t j, hj;
419
420 // AXPY algorithm.
421 for (Ak = A.seg_begin(); Ak < Aend; ++Ak)
422 {
423 j = Ak.index();
424 hj = Ak.size();
425 if (hj == 0) continue;
426 Aij = Ak.begin();
427 vecvec::axpy_n(x[j] * scale, Aij, y, hj);
428 }
429}

434__mult (T scale, const Matrix& A,

473// Symmetric matrix, any orientation, full, non-blocked.
474template <class T, class Matrix, class IterX, class IterY>
475void
476__mult (T scale, const Matrix& A,
477 cm2::symmetric_tag, typename Matrix::orientation_tag, cm2::full_tag, cm2::non_blocked_tag,
478 IterX x, IterY y, int /*nthreads*/)
479{
480 const typename Matrix::const_segment_type Aend(A.seg_end());
481 typename Matrix::const_segment_type Ak;
482
483 for (Ak = A.seg_begin(); Ak < Aend; ++Ak)
484 {
485 const size_t j(Ak.index()), ibeg(Ak.index_begin()), hj(Ak.size());
486 if (hj == 0) continue;
487 y[j] += scale * vecvec::dot_n(Ak.begin(), x+ibeg, hj);
488 vecvec::axpy_n(x[j] * scale, Ak.begin(), y+ibeg, std::min(hj,j-ibeg)); // min(hj,j-ibeg) to avoid diagonal processed twice.
489 }
490}

492// Symmetric matrix, any orientation, sparse, non-blocked.
493template <class T, class Matrix, class IterX, class IterY>
494void
495__mult (T scale, const Matrix& A,
496 cm2::symmetric_tag, typename Matrix::orientation_tag, cm2::sparse_tag, cm2::non_blocked_tag,
497 IterX x, IterY y, int /*nthreads*/)
498{
499 const typename Matrix::const_segment_type Aend(A.seg_end());
500 typename Matrix::const_segment_type Ak;
501 typename Matrix::const_segment_type::iterator Aij;
502 size_t j, hj;
503
504 for (Ak = A.seg_begin(); Ak < Aend; ++Ak)
505 {
506 j = Ak.index();
507 hj = Ak.size();
508 if (hj == 0) continue;
509 Aij = Ak.begin();
510 y[j] += scale * vecvec::dot_n(Aij, x, hj);
511 if (Aij.index() == j) // To avoid diagonal processed twice.
512 { ++Aij; --hj; }
513 vecvec::axpy_n(x[j] * scale, Aij, y, hj);
514 }
515}

537template <class T, class BlockIterator, class IterX, class IterY>
538INLINE void
539__mult_blocks (T scale, BlockIterator A0, int nbr_blocks,
540 cm2::non_symmetric_tag, cm2::upper_right_storage_tag,
541 cm2::separate_diagonal_tag /*BlockDiagonalTag*/,
542 IterX x, IterY y, int nthreads)
543{
544 typedef typename BlockIterator::value_type block_type;
545 typedef typename block_type::storage_tag storage_tag;
546 typedef typename block_type::orientation_tag orientation_tag;
547 typedef typename block_type::fullness_tag fullness_tag;
548 typedef typename block_type::blocking_tag block_blocking_tag;
549 typedef typename cm2::orientation_as_upper<storage_tag, orientation_tag>::tag ur_orientation_tag;
550
551 BlockIterator Ak;
552
553 #pragma omp parallel for private(Ak) schedule(dynamic, 1) num_threads(nthreads)
554 for (int k = 0; k < nbr_blocks; ++k) // Parallel loop over in-core blocks.
555 {
556 Ak = A0; Ak += k;
557 const size_t i0(Ak->first_index());
558 const size_t n(Ak->dim());
559
560 if (Ak->segments_entries() > 0)
561 matvec::__mult(scale, *Ak,
562 cm2::non_symmetric_tag(), ur_orientation_tag(), fullness_tag(), block_blocking_tag(),
563 x, y, /*nthreads=>*/ 1);
564
565 if (Ak->is_unit_diagonal())
566 vecvec::axpy_n(scale, x+i0, y+i0, n);
567 else
568 vecvec::axypz_n(scale, Ak->diagonal_begin(), x+i0, y+i0, n);
569 }
570}

644// Blocked matrix (any shape, any orientation).
645template <class T, class Matrix, class IterX, class IterY>
646INLINE void
647__mult (T scale, const Matrix& A,
648 typename Matrix::shape_tag, typename Matrix::orientation_tag, typename Matrix::fullness_tag, cm2::blocked_tag,
649 IterX x, IterY y, int nthreads)
650{
651 const size_t NB(A.blocks());
652 typename Matrix::const_block_iterator A0(A.blocks_begin());
653 size_t nb;
654
655 for (size_t B0 = 0; B0 < NB; B0 += nb, A0 += nb)
656 {
657 (void)A0->data(); // Load A0 and following blocks.
658 nb = A.incore_blocks();
659 assert(nb <= NB - B0);
660
661#ifdef _OPENMP
662 const size_t se(A.incore_entries());
663 const bool parallel_cond = (nthreads > 1) && (nb > 1) && (se > CM2_OMP_MIN_LOOP);
664 if (parallel_cond)
665 __mult_blocks(scale, A0, nb,
666 typename Matrix::shape_tag(), typename Matrix::block_type::diagonal_storage_tag(),
667 x, y, nthreads);
668 else
669#endif
670 {
671 typename Matrix::const_block_iterator Ak(A0);
672 for (size_t k = 0; k < NB; ++k, ++Ak)
673 matvec::__mult(scale, *Ak, x, y, nthreads);
674 }
675 }
676}
