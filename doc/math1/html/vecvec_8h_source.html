183template <class T>
184INLINE void
185swap_n (T* CM2_RESTRICT x, T* CM2_RESTRICT y, size_t n)
186{
187#ifdef CM2_NO_UNROLL
188 for (size_t i = 0; i < n; ++i)
189 std::swap(x[i], y[i]);
190#else
191 if (n >= 4)
192 {
193 const size_t n4(n / 4);
194 n = n % 4;
195 for (size_t i = 0; i < n4; ++i)
196 {
197 std::swap(x[0], y[0]);
198 std::swap(x[1], y[1]);
199 std::swap(x[2], y[2]);
200 std::swap(x[3], y[3]);
201 x += 4;
202 y += 4;
203 }
204 }
205
206 if (n == 1)
207 y[0] = x[0];
208 else if (n == 2)
209 vecvec::swap2(x, y);
210 else if (n == 3)
211 vecvec::swap3(x, y);
212#endif
213}
