56template <class IStream, class Matrix>
57int
58read (IStream& s, Matrix& A)
59{
60 typedef typename Matrix::value_type value_type;
61
62 value_type v;
63 char c(' ');
64 size_t M, N; // Matrix dimensions
65
66 A.clear();
67
68 do { s.get(c); } while (s.good() && ::isspace(int(c)));
69 if (!s.good()) return -1; // error
70 if (::isdigit(int(c))) s.putback(c);
71 s >> M;
72 if (!s.good()) return -1; // error
73
74 do { s.get(c); } while (s.good() && ::isspace(int(c)));
75 if (!s.good()) return -1; // error
76 if (::isdigit(int(c))) s.putback(c);
77 s >> N;
78 if (!s.good()) return -1; // error
79
80 do { s >> c; } while (s.good() && ::isspace(int(c))); // Eat the leading '[' if it exists
81 if (!s.good()) return -1; // error
82 if (c != '[') { s.putback(c); }
83
84 if (!A.resize(M, N))
85 return -199; // Memory error.
86
87 for (size_t i = 0; (i < M) && s.good(); i++)
88 {
89 for (size_t j = 0; (j < N) && s.good(); j++)
90 {
91 s >> v;
92 A(i, j) = v;
93 }
94 }
95
96 if (s.good())
97 {
98 do { s.get(c); } while (s.good() && ::isspace(int(c))); // Eat the trailing ']'.
99 }
100 if (s.good() && (c != ']')) { s.putback(c); }
101
102 return s.good() ? 0 : -1;
103}
