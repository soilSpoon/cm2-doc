461bool
462copy (const self_type& x)
463{
464 bool ok(true);
465
466 if (this == &x)
467 return ok;
468
469 if (this->__rep() == x.__rep())
470 this->shallow_copy(self_type()); // New buffer.
471
472 this->clear();
473 ok = this->resize(x.rows(), x.cols());
474 if (ok)
475 matmat::copy(x, *this);
476 return ok;
477}

1080bool
1081reserve (size_type m, size_type n, size_t growth_rate = 1)
1082{
1083 const size_type m0(this->_rows), n0(this->_cols);
1084
1085 // Reserve new columns with growth_rate.
1086 if ((m == m0) && (m == this->_ld) && (n > n0) && (growth_rate > 1))
1087 return parent_type::reserve(m * n, growth_rate);
1088
1089 else if ((m > m0) || (n > n0))
1090 return this->resize(m, n) && this->resize(m0, n0);
1091
1092 return true;
1093}

1108{
1109 // If one dimension is null, nullify the two.
1110 if (m * n == 0)
1111 m = n = 0;
1112
1113 // Fast return.
1114 if ((m == this->_rows) && (n == this->_cols))
1115 return true;
1116
1117 // Try to avoid a reallocation.
1118 const size_type ld(this->empty() ? m : this->_ld);
1119 if ((m <= ld) && (ld * n <= parent_type::capacity()))
1120 {
1121 parent_type::resize(ld * n);
1122 this->_rows = m;
1123 this->_cols = n;
1124 this->_ld = ((m != 0) ? ld : 0);
1125 return true;
1126 }
1127
1128 // Reallocation compulsory.
1129 self_type tmp(m, n);
1130 pointer p1(this->data()), p2(tmp.data());
1131 const size_type m1(std::min(this->_rows, m));
1132 const size_type n1(std::min(this->_cols, n));
1133
1134 if ((tmp.rows() != m) || (tmp.cols() != n))
1135 return false;
1136
1137 assert(tmp.ld() == m);
1138 if ((this->_ld == m1) && (m == m1))
1139 vecvec::copy_n(p1, p2, m1*n1);
1140 else
1141 {
1142 for (size_type j = 0; j < n1; ++j, p1 += this->_ld, p2 += m)
1143 vecvec::copy_n(p1, p2, m1);
1144 }
1145 (*this) = tmp;
1146
1147 return true;
1148}

1161bool
1162resize (size_type m, size_type n, value_type v)
1163{
1164 const size_type m0(this->_rows), n0(this->_cols);
1165 const bool ok(this->resize(m, n));
1166 typename self_type::iterator p;
1167
1168 if (this->empty() || !ok)
1169 return ok;
1170
1171 if ((m > m0) && (n0 > 0))
1172 {
1173 p = this->begin() + m0;
1174 for (size_type j = 0; j < n0; ++j, p += this->_ld)
1175 vecscal::copy_n(v, p, m - m0);
1176 }
1177
1178 if (n > n0)
1179 {
1180 p = this->begin() + n0 * this->_ld;
1181 if (this->_ld == m)
1182 vecscal::copy_n(v, p, m * (n - n0));
1183 else
1184 {
1185 for (size_type j = n0; j < n; ++j, p += this->_ld)
1186 vecscal::copy_n(v, p, m);
1187 }
1188 }
1189
1190 return true;
1191}

1313

1349 s.write(reinterpret_cast<const char_type*>(&_cols), sizeof(size_t) / sizeof(char_type)); // Write _cols.

1352 parent_type::save(s);
