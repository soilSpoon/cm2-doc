38// Reads a dynamic-sized vector from an input stream.
39template <class IStream, class Vector>
40int
41__read (IStream& s, Vector& V, cm2::dynamic_size_tag)
42{
43 typedef typename Vector::value_type value_type;
44
45 value_type v;
46 char c(' ');
47 size_t N(0);
48
49 V.clear();
50
51 // Skip any white space.
52 do { s.get(c); } while (s.good() && ::isspace(int(c)));
53
54 // Read size (optional).
55 if (::isdigit(int(c)) && s.good())
56 {
57 s.putback(c);
58 s >> N;
59 V.reserve(N);
60
61 // Skip white spaces.
62 do { s.get(c); } while (s.good() && ::isspace(int(c)));
63 }
64
65 if (s.good() && (c != '['))
66 {
67 s.putback(c);
68 return -1; // error.
69 }
70
71 // Skip white spaces.
72 do { s.get(c); } while (s.good() && ::isspace(int(c)));
73
74 while (s.good() && (c != ']'))
75 {
76 s.putback(c);
77 if (s >> v)
78 V.push_back(v);
79
80 // Skip white spaces.
81 do { s.get(c); } while (s.good() && ::isspace(int(c)));
82 }
83
84 return s.good() ? int(V.size()) : -1;
85}

87// Reads a fixed-sized vector from an input stream.
88template <class IStream, class Vector, class NonDynamicTag>
89int
90__read (IStream& s, Vector& V, NonDynamicTag)
91{
92 typedef typename Vector::value_type value_type;
93 typedef typename Vector::iterator iterator;
94
95 value_type v;
96 iterator p(V.begin()), pend(V.end());
97 char c(' ');
98 size_t N(V.size());
99
100 // Skip any white space.
101 do { s.get(c); } while (s.good() && ::isspace(int(c)));
102
103 // Read size (optional).
104 if (::isdigit(int(c)) && s.good())
105 {
106 s.putback(c);
107 s >> N;
108 do { s.get(c); } while (s.good() && ::isspace(int(c)));
109 }
110
111 if (s.good() && (c != '['))
112 {
113 s.putback(c);
114 return -1; // error.
115 }
116
117 if (N != V.size())
118 return -2; // Size error.
119
120 // Skip white spaces.
121 do { s.get(c); } while (s.good() && ::isspace(int(c)));
122
123 while (s.good() && (p < pend) && (c != ']'))
124 {
125 s.putback(c);
126 if (s >> v)
127 *p++ = v;
128
129 // Skip white spaces.
130 do { s.get(c); } while (s.good() && ::isspace(int(c)));
131 }
132
133 return s.good() ? int(p - pend) : -1;
134}

182template <class Ostream, class Vector>
183void
184write (Ostream& s, const Vector& V, size_t n = UINT_MAX)
185{
186 size_t N(V.size());
187 typename Vector::const_iterator p(V.begin());
188
189 s << int(N) << "[";
190 while ((N >= n) && s.good())
191 {
192 N -= n;
193 for (size_t k = 0; (k < n) && s.good(); ++k)
194 s << *p++ << " ";
195 s << "\n";
196 }
197 while ((N-- > 0) && s.good())
198 s << *p++ << " ";
199 s << "]\n";
200}
