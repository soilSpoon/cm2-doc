838load (IStream& s)
839{
840 typedef typename IStream::char_type char_type;
841
842 int ret(0);
843
844 this->clear();
845

914size_type
915segments_fitting_in (size_type sz, size_type seg0, size_type& se) const
916{
917 assert(seg0 <= this->_dim);
918
919 size_type dseg(this->_dim - seg0);
920 const size_t c(2 * dseg + 1);
921
922 sz /= sizeof(T);
923
924 if (4 * sz < c*c)
925 {
926 const float d = static_cast<float>(c*c - 4*sz);
927 dseg = (c - static_cast<size_t>(::sqrt(d))) / 2;
928 }
929
930 se = (dseg * (c - dseg)) / 2;
931 return dseg;
932}
