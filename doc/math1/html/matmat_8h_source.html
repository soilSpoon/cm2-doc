63template <class Mat1, class Mat2>
64bool
65push_back (const Mat1& M1, Mat2& M2)
66{
67 // Fast return.
68 if (M1.empty())
69 return true;
70
71 const size_t R(M1.rows()), C1(M1.cols()), LD1(M1.ld());
72 const size_t C2(M2.cols());
73 typename Mat1::const_iterator p1(M1.begin());
74 typename Mat2::iterator p2;
75 size_t ld2;
76
77 assert(M2.empty() || (M1.rows() == M2.rows()));
78
79 if (C2 + C1 > M2.cols_capacity())
80 M2.reserve(R, C2 + std::max(C1, C2)); // At least double the capacity of the matrix.
81
82 if (!M2.resize(R, C2 + C1))
83 return false; // Memory error.
84
85 ld2 = M2.ld();
86 p2 = M2.begin() + ld2 * C2;
87
88 if ((LD1 == R) && (ld2 == R)) // Contiguous storage.
89 vecvec::copy_n(p1, p2, R * C1);
90 else
91 {
92 for (size_t j = 0; j < C1; ++j, p1 += LD1, p2 += ld2)
93 vecvec::copy_n(p1, p2, R);
94 }
95
96 return true; // OK.
97}

118template <class Mat1Iterator, class Mat2>
119bool
120push_back (Mat1Iterator M1_beg, Mat1Iterator M1_end, Mat2& M2)
121{
122 // Fast return.
123 if (M1_beg == M1_end)
124 return true;
125
126 size_t R(M2.rows()), C2(M2.cols());
127 Mat1Iterator pm;
128
129 // Check dimensions.
130 for (pm = M1_beg; pm < M1_end; ++pm)
131 {
132 if (pm->empty()) continue;
133 C2 += pm->cols();
134 if (R == 0) R = pm->rows();
135 if (pm->rows() != R)
136 return false; // Dimension error.
137 }
138
139 if (!M2.reserve(R, C2))
140 return false; // Memory error.
141
142 for (pm = M1_beg; pm < M1_end; ++pm)
143 M2.push_back(*pm);
144
145 return true; // OK.
146}

163template <class Mat1, class Mat2, class VecI>
164bool
165push_back (const Mat1& M1, const VecI& M1_cols, Mat2& M2)
166{
167 // Fast return.
168 if (M1_cols.empty())
169 return true;
170
171 const size_t R(M1.rows()), C2(M2.cols());
172 const size_t NI(M1_cols.size());
173 typename Mat2::iterator p2;
174 size_t ld2, j_;
175
176 assert(vecscal::max_value(M1_cols) < M1.cols());
177 assert(M2.empty() || (M1.rows() == M2.rows()));
178
179 if (C2 + NI > M2.cols_capacity())
180 M2.reserve(R, C2 + std::max(C2, NI)); // At least double the capacity of the matrix.
181
182 if (!M2.resize(R, C2 + NI))
183 return false; // Memory error.
184
185 ld2 = M2.ld();
186 p2 = M2.begin() + ld2 * C2;
187
188 for (size_t j = 0; j < NI; ++j, p2 += ld2)
189 {
190 j_ = M1_cols[j];
191 vecvec::copy_n(M1.col_begin(j_), p2, R);
192 }
193
194 return true; // OK.
195}

246template <class Mat1, class VecI, class Mat>
247bool
248insert (const Mat1& M1, const VecI& M_cols, Mat& M)
249{
250 // Fast return.
251 if (M1.empty())
252 return true;
253
254 const size_t N0(M.cols()), N1(M1.cols());
255 const size_t R(M.rows());
256 typename Mat1::const_segment_iterator seg1(M1.seg_end() - 1);
257 typename VecI::const_reverse_iterator pi(M_cols.rbegin());
258 typename Mat::segment_iterator seg0, seg0_;
259 size_t iprev(N0 + N1), s(0);
260
261 assert(M1.cols() == M_cols.size());
262 assert(M.rows() == M1.rows());
263#if defined(_MSC_VER) || (__cplusplus >= 201100)
264 assert(std::is_sorted(M_cols.begin(), M_cols.end()));
265#endif
266 assert(std::adjacent_find(M_cols.begin(), M_cols.end()) == M_cols.end());
267 assert(M_cols.back() < N0 + N1);
268
269 if (!M.resize(R, N0 + N1))
270 return false; // Memory error.
271
272 seg0_ = M.seg_end() - 1;
273 seg0 = seg0_ - N1;
274 for (size_t k = 0; k < N1; ++k) // Backward loop.
275 {
276 s = iprev - (*pi) - 1;
277 iprev = *pi++;
278 for (size_t n = 0; n < s; ++n, --seg0, --seg0_)
279 vecvec::copy_n(seg0.begin(), seg0_.begin(), R);
280 vecvec::copy_n(seg1.begin(), seg0_.begin(), R);
281 --seg0_; --seg1;
282 }
283
284 return true; // OK.
285}

298template <class VecI, class Mat>
299void
300remove (const VecI& M_cols, Mat& M)
301{
302 // Fast return.
303 if (M_cols.empty())
304 return;
305
306 const size_t C(M.cols()), R(M.rows());
307 const typename VecI::const_iterator pjend(M_cols.end());
308 typename VecI::const_iterator pj(M_cols.begin());
309 typename Mat::segment_type segj(M.seg(*pj)), segj2(segj);
310
311#if defined(_MSC_VER) || (__cplusplus >= 201100)
312 assert(std::is_sorted(pj, pjend));
313#endif
314 assert(std::adjacent_find(M_cols.begin(), M_cols.end()) == M_cols.end());
315 assert(M_cols.back() < C);
316
317 ++segj; ++pj; // Remove this first column.
318
319 while (segj.index() < C)
320 {
321 if ((pj != pjend) && (*pj == segj.index())) // Remove this column.
322 ++pj;
323 else
324 {
325 vecvec::copy_n(segj.begin(), segj2.begin(), R);
326 ++segj2;
327 }
328
329 ++segj;
330 }
331
332 M.resize(M.rows(), segj2.index());
333}

340template <class Mat1, class Mat2>
341bool
342equal (const Mat1& M1, const Mat2& M2)
343{
344 const size_t C(M1.cols()), R(M1.rows());
345
346 if ((M1.rows() != M2.rows()) || (M1.cols() != M2.cols()))
347 return false; // Dimensions are different.
348
349 if ((M1.ld() == R) && (M2.ld() == R)) // Contiguous storage.
350 return vecvec::equal_n(M1.begin(), M2.begin(), R * C);
351 else
352 {
353 typename Mat1::const_segment_type c1(M1.seg_begin());
354 typename Mat2::const_segment_type c2(M2.seg_begin());
355
356 for (size_t j = 0; j < C; ++j, ++c1, ++c2)
357 if (!vecvec::equal_n(c1.begin(), c2.begin(), R))
358 return false;
359 }
360
361 return true;
362}

1314 A, B, C non-symmetric, full, col-oriented.
1315 \note Uses hierarchic block + cache tiling.
1316 */
1317template <class T, class MatA, class MatB, class MatC>
1318void
1319__mult
1320 (T a,
1321 const MatA& A, cm2::non_symmetric_tag, cm2::full_tag, cm2::col_oriented_tag,
1322 const MatB& B, cm2::non_symmetric_tag, cm2::full_tag, cm2::col_oriented_tag,
1323 MatC& C, cm2::non_symmetric_tag, cm2::full_tag, cm2::col_oriented_tag,
1324 size_t cache_size, int nthreads)
1325{
1326 typedef typename MatA::value_type TA;
1327 typedef typename MatB::value_type TB;
1328 typedef typename MatC::value_type TC;
1329
1330 const size_t ROWS(A.rows());
1331 const size_t COLS(B.cols());
1332 const size_t ACOLS(A.cols());
1333 const typename MatA::const_segment_type A_0(A.seg_begin());
1334 const typename MatB::const_segment_type B_0(B.seg_begin());
1335 const typename MatC::segment_type C_0(C.seg_begin());
1336 typename MatA::const_segment_type A_I0, A_I1;
1337 typename MatB::const_segment_type B_J0, B_J1;
1338 typename MatC::segment_type C_J0, C_J1;
1339 const TB* CM2_RESTRICT pb;
1340 TC* CM2_RESTRICT pc;
1341 size_t Ci(ACOLS), Cj(COLS), ci, cj, ni, nj;
1342 size_t I0, I1, J0, J1;
1343
1344 assert(A.rows() == C.rows());
1345 assert(B.cols() == C.cols());
1346
1347 // Compute the number of columns fitting in cache.
1348 {
1349 const size_t a_ = A.rows() * sizeof(TA);
1350 const size_t b_ = B.rows() * sizeof(TB);
1351 const size_t c_ = C.rows() * sizeof(TC);
1352 if ((cache_size > 0) && (cache_size < Ci*a_ + Cj*(b_+c_))) // Doesn't fit in cache.
1353 {
1354 const double r = double(cache_size) / double(Ci*a_ + Cj*(b_+c_));
1355 Ci = std::max(size_t(r * double(ACOLS)), size_t(1));
1356 Cj = std::max(size_t(r * double(COLS)), size_t(1));
1357 }
1358 }
1359
1360 // Loop on blocks of B-C.
1361 for (J0 = 0, B_J0 = B_0, C_J0 = C_0; J0 < COLS; J0 += nj, B_J0 += nj, C_J0 += nj)

1364 pc = C_J0.begin(); // Load first cols of C.
1365 assert(B.last_incore_col() > J0);
1366 assert(C.last_incore_col() > J0);
1367 nj = std::min(B.last_incore_col(), C.last_incore_col()) - J0;

1371 {

1373 assert(A.last_incore_col() > I0);
1374 ni = A.last_incore_col() - I0;
1375
1376 // Loop on cache blocks of B-C.
1377 for (J1 = J0, B_J1 = B_J0, C_J1 = C_J0; J1 < J0 + nj; J1 += cj, B_J1 += cj, C_J1 += cj)
1378 {

1384 {
1385 ci = std::min(Ci, ni);
1386 ci = std::min(ci, I0 + ni - I1);

1388 // Loop inside B-C block.
1389#ifdef _OPENMP

1391 {
1392 #pragma omp parallel for private(pb,pc) schedule(static) num_threads(nthreads) // Parallelization over columns.
1393 for (int j = 0; j < int(cj); ++j)

1395 typename MatA::const_segment_type A_i(A_I1);
1396 typename MatB::const_segment_type B_j(B_J1); B_j += j;
1397 typename MatC::segment_type C_j(C_J1); C_j += j;
1398
1399 pb = B_j.begin() + I1; pc = C_j.begin();
1400 for (size_t i = 0; i < ci; ++i, ++pb, ++A_i)
1401 vecvec::axpy_n(a * (*pb), A_i.begin(), pc, ROWS);
1402 }
1403 }
1404 else
1405#endif
1406 {
1407 typename MatA::const_segment_type A_i(A_I1);
1408 typename MatB::const_segment_type B_j(B_J1);
1409 typename MatC::segment_type C_j(C_J1);
1410
1411 for (size_t j = 0; j < cj; ++j, ++B_j, ++C_j, A_i = A_I1)
1412 {
1413 pb = B_j.begin() + I1; pc = C_j.begin();
1414 for (size_t i = 0; i < ci; ++i, ++pb, ++A_i)
1415 vecvec::axpy_n_par(a * (*pb), A_i.begin(), pc, ROWS, nthreads); // Parallelization inside columns.
1416 }
1417 }
1418 }
1419 }
1420 }

1510 vecvec::axpy_n(a * (*pb), A_i.begin(), pc, J1+j+1);

1515 {
1516 typename MatA::const_segment_type A_i(A_I1);
1517 typename MatB::const_segment_type B_j(B_J1);
1518 typename MatC::segment_type C_j(C_J1);

1521 {
1522 pb = B_j.begin() + I1; pc = C_j.begin();
1523 for (size_t i = 0; i < ci; ++i, ++pb, ++A_i)
1524 vecvec::axpy_n_par(a * (*pb), A_i.begin(), pc, J1+j+1, nthreads); // Parallelization inside columns.

1526 }
1527 }
