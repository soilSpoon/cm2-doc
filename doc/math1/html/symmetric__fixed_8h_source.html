746size_type
747segments_fitting_in (size_type sz, size_type seg0, size_type& se) const
748{
749 assert(seg0 <= N);
750
751 size_type dseg(N - seg0);
752 const size_t c(2 * dseg + 1);
753
754 sz /= sizeof(T);
755
756 if (4 * sz < c * c)
757 {
758 const float df = static_cast<float>(c * c - 4 * sz);
759 dseg = (c - static_cast<size_t>(::sqrt(df))) / 2;
760 }
761
762 se = (dseg*(c-dseg)) / 2;
763 return dseg;
764}
