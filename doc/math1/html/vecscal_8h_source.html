1452template <class IterX, class T>
1453void
1454statistics_n (IterX x, size_t n, T& mean, T& variance)
1455{
1456 const T inv_n = 1. / T(n);
1457
1458 mean = variance = T(0);
1459
1460 for (size_t i = 0; i < n; ++i, ++x)
1461 {
1462 mean += (*x);
1463 variance += (*x) * (*x);
1464 }
