378template <class SparseIterator>
379inline void
380axpy_col (T a, size_t j, SparseIterator first, SparseIterator last)
381{
382 const iterator p0(this->col_begin(j));
383 size_t i;
384
385 for (SparseIterator ps = first; ps != last; ++ps)
386 {
387 i = ps.index();
388 assert(i < this->rows());
389 *(p0 + i) += a * (*ps);
390 }
391}
