627template <class Index_>
628reference
629operator() (Index_ i, Index_ j)
630{
631 typename indices_array_type::const_iterator pj, pjend;
632
633 if (i > j) std::swap(i, j); // i <= j.
634 assert(size_type(j) < this->dim());
635 pj = this->_ja.begin() + this->_ia[i];
636 pjend = this->_ja.begin() + this->_ia[i+1];
637 if (*pj != static_cast<index_type>(j))
638 pj = std::lower_bound(pj + 1, pjend, static_cast<index_type>(j));
639 assert(*pj == static_cast<index_type>(j));
640 return parent_type::operator[](pj - this->_ja.begin());
641}

671template <class SparseIterator>
672void
673axpy_row (T a, size_t i, SparseIterator first, SparseIterator last)
674{
675 typename indices_array_type::iterator pj, pjend;
676 index_type j;
677
678#if defined(_MSC_VER) || (__cplusplus >= 201100)
679 assert(std::is_sorted(first.idx_iter(), last.idx_iter()));
680#endif
681 pj = this->_ja.begin() + this->_ia[i];
682 pjend = this->_ja.begin() + this->_ia[i+1];
683
684 for (SparseIterator ps = first; ps != last; ++ps, ++pj)
685 {
686 j = static_cast<index_type>(ps.index());
687 assert(i <= j);
688 if (*pj < j) ++pj;
689 if (*pj < j) pj = std::lower_bound(pj, pjend, j);
690 assert(*pj == j);
691 parent_type::operator[](pj - this->_ja.begin()) += a * (*ps);
692 }
693}

770template <class AVector, class IVector, class JVector>
771static bool
772is_valid (const AVector& A, const IVector& ia, const JVector& ja)
773{
774 const size_type DIM(ia.size() - 1); // Dimension of the matrix.
775 typename JVector::const_iterator pj, pjend(ja.begin());
776 size_t j(0);
777
778 if (A.empty() && ia.empty() && ja.empty())
779 return true; // Accept when all arrays are empty.
780
781 if (ia.empty()) return false; // ia must contains at least one element (0) for an empty matrix.
782 if (ia.front() != 0) return false; // ia must start with 0.
783 if (ia.back() != A.size()) return false; // Last value in ia must point to past-the-end in A and ja.
784 if (A.size() != ja.size()) return false; // A and ja must have same size.
785
786 for (size_type i = 0; i < DIM; ++i)
787 {
788 pj = pjend;
789 pjend = ja.begin() + ia[i+1];
790 if (pjend < pj)
791 return false; // ia must be sorted ascendingly.
792 if (pjend == pj)
793 continue;
794 j = *pj;
795 if (j < i)
796 return false; // Symmetric upper-right.
797 while (++pj < pjend)
798 {
799 if (*pj <= j)
800 return false; // ja must be sorted ascendingly per row.
801 j = *pj;
802 }
803 if (j >= DIM)
804 return false;
805 }
806
807 return true; // OK.
808}
