589bool

591{
592 if (n > this->capacity())
593 this->_rep->reserve(this->_shift + n,
594 /*recopy_up_to=>*/ this->_shift + this->_size,

596

700template <class IStream>
701int
702load (IStream& s)
703{
704 typedef typename IStream::char_type char_type;
705
706 size_t sz(0);
707
708 this->clear();
709
710 s.read(reinterpret_cast<char_type*>(&sz), sizeof(size_t) / sizeof(char_type)); // Read size.
711 if (!s.good())
712 return -1;
713 if (!this->resize(sz))
714 return -199; // Memory error.
715
716 s.read(reinterpret_cast<char_type*>(this->begin()), this->size_of() / sizeof(char_type)); // Read data.
717
718 return s.good() ? 0 : -1;
719}

731save (OStream& s) const

734

736
737 s.write(reinterpret_cast<const char_type*>(&sz), sizeof(size_t) / sizeof(char_type)); // Write size.
738 s.write(reinterpret_cast<const char_type*>(this->begin()), this->size_of() / sizeof(char_type)); // Write data.
739}
